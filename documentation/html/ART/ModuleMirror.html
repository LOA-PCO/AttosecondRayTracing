ART.ModuleMirror

Provides classes for different mirror surfaces, which are types of optical elements. Think of these as the z-coordinates on top of the x-y-grid provided by the support.

Also provides the function ReflectionMirrorRayList that returns the rays reflected on the mirror. Rays that do not hit the support are not further propagated.

 32    """
 33    A plane mirror.
 34    
 35    Attributes
 36    ----------
 37        support : Support-object from ModuleSupport
 38        
 39        type : 'Plane Mirror'.
 40            
 41    Methods
 42    ----------
 43        get_normal(Point)
 44        
 45        get_centre()
 46        
 47        get_grid3D(NbPoints)
 48    """
 49    def __init__(self, Support):
 50        """
 51        Parameters
 52        ----------
 53            Support : Support-object from ModuleSupport
 54
 55        """
 56        self.support = Support
 57        self.type = 'Plane Mirror'
 58
 59    def _get_intersection(self, Ray):
 60        """  Return the intersection point between Ray and the xy-plane  """
 61        t = -Ray.point[2] / Ray.vector[2]
 62        I = Ray.vector*t + Ray.point
 63        if t > 0 and self.support._IncludeSupport(I):
 64            PointIntersection = Ray.vector*t + Ray.point
 65        else: 
 66            PointIntersection = None
 67
 68        return PointIntersection
 69

 71    def get_normal(self, Point : np.ndarray):
 72        """  Return normal unit vector in point 'Point' on the plane mirror. """
 73        Normal = np.array([0,0,1])
 74        return Normal
 75    
 76    def get_centre(self):
 77        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
 78        return np.array([0,0,0])

 81        """
 82        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,

 87        for k in ListCoordXY:
 88            z = 0

 94class MirrorSpherical:
 95    """
 96    Spherical mirror surface with eqn. $x^2 + y^2 + z^2  = R^2$, where $R$ is the radius.
 97    
 98    ![Illustration of a spherical mirror.](../documentation/sphericalmirror.svg)
 99    
100    Attributes
101    ----------
102        radius : float
103            Radius of curvature. A postive value for concave mirror, a negative value for a convex mirror.
104    
105        support : Support-object from ModuleSupport
106        
107        type : str SphericalCC Mirror' or SphericalCX Mirror'
108            
109    Methods
110    ----------
111        get_normal(Point)
112        
113        get_centre()
114        
115        get_grid3D(NbPoints)
116    
117    """
118    def __init__(self, Radius, Support):
119        """
120        Parameters
121        ----------
122            Radius : float
123                The radius of curvature in mm. A postive value for concave mirror, a negative value for a convex mirror.
124            
125            Support : Support-object from ModuleSupport
126
127        """
128        if Radius <0:
129            self.type = 'SphericalCX Mirror'
130            self.radius = -Radius
131        else:
132            self.type = 'SphericalCC Mirror'
133            self.radius = Radius
134        
135        self.support = Support
136        
137    
138    def _get_intersection(self, Ray):
139        """  Return the intersection point between the ray and the sphere  """                
140        a = np.dot(Ray.vector,Ray.vector)
141        b = 2 * np.dot(Ray.vector, Ray.point)
142        c = np.dot(Ray.point, Ray.point) - self.radius**2
143        
144        Solution = mgeo.SolverQuadratic(a,b,c)
145        Solution = mgeo.KeepPositiveSolution(Solution)
146        
147        ListPointIntersection = []
148        for t in Solution:
149            I = Ray.vector*t + Ray.point
150            if I[2] < 0 and self.support._IncludeSupport(I):
151                ListPointIntersection.append(I) 
152                    
153        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
154    
155    def get_normal(self, Point):
156        """ Return the normal unit vector on the spherical surface at point Point """
157        Gradient = Point
158        return mgeo.Normalize(Gradient)
159    
160    def get_centre(self):
161        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
162        return np.array([0,0,-self.radius])
163    
164    def get_grid3D(self,NbPoint):
165        """
166        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
167        sampling the support in a number NbPoints of points.
168        """
169        ListCoordXYZ = []
170        ListCoordXY = self.support._get_grid(NbPoint)
171        for k in ListCoordXY:
172            z = -np.sqrt(self.radius**2 - (k[0]**2 + k[1]**2))
173            ListCoordXYZ.append(np.array([k[0],k[1],z]))
174        return ListCoordXYZ

178class MirrorParabolic:
179    """ 
180    A paraboloid with vertex at the origin [0,0,0] and symmetry axis z, 
181    with the center of the support shifted off axis along the x-direction.
182    Its eqn. is therefore $z = \\frac{1}{2p}[(x+x_c)^2 + y^2]$ where $p$ is the semi latus rectum,
183    equal to twice the focal lenght of the *mother* parabola (i.e. off-axis angle $\\alpha=0$).
184    
185    The effective focal length $f_\\mathrm{eff}$ is related to the semi latus rectum by 
186    $ p = f_\\mathrm{eff} (1+\\cos\\alpha)$, where $\\alpha$ is the off-axis angle.
187    
188    ![Illustration of a parabolic mirror.](../documentation/parabola.svg)
189
190    Attributes
191    ----------
192        offaxisangle : float
193            Off-axis angle of the parabola. Modifying this also updates p, keeping feff constant.
194            Attention: The off-axis angle must be *given in degrees*, but is stored and will be *returned in radian* !
195            
196        feff : float
197            Effective focal length of the parabola in mm. Modifying this also updates p, keeping offaxisangle constant.
198    
199        p : float
200            Semi latus rectum of the parabola in mm. Modifying this also updates feff, keeping offaxisangle constant.
201    
202        support : Support-object from ModuleSupport
203        
204        type : str 'Parabolic Mirror'.
205            
206    Methods
207    ----------
208        get_normal(Point)
209        
210        get_centre()
211        
212        get_grid3D(NbPoints)
213    
214    """
215    def __init__(self, FocalEffective :float, OffAxisAngle :float, Support):
216        """
217        Parameters
218        ----------
219            FocalEffective : float
220                Effective focal length of the parabola in mm.
221            
222            OffAxisAngle : float
223                Off-axis angle *in degrees* of the parabola.
224            
225            Support : Support-object from ModuleSupport
226
227        """
228        self._offaxisangle = np.deg2rad(OffAxisAngle) 
229        self.support = Support
230        self.type = 'Parabolic Mirror'
231        self._feff = FocalEffective  #effective focal length
232        self._p = FocalEffective *(1+np.cos(self.offaxisangle)) #semi latus rectum
233    
234    @property
235    def offaxisangle(self): 
236        return self._offaxisangle
237       
238    @offaxisangle.setter 
239    def offaxisangle(self, OffAxisAngle): 
240        self._offaxisangle = np.deg2rad(OffAxisAngle)
241        self._p = self._feff *(1+np.cos(self._offaxisangle)) #make sure to always update p
242        
243    @property
244    def feff(self): 
245        return self._feff
246       
247    @feff.setter 
248    def feff(self, FocalEffective): 
249        self._feff = FocalEffective
250        self._p = self._feff *(1+np.cos(self.offaxisangle)) #make sure to always update p
251        
252    @property
253    def p(self): 
254        return self._p
255       
256    @p.setter 
257    def p(self, SemiLatusRectum): 
258        self._p = SemiLatusRectum
259        self._feff = self._p/(1+np.cos(self.offaxisangle))  #make sure to always update feff
260 
261    
262    def _get_intersection(self, Ray):
263        """  Return the intersection point between the ray and the parabola  """
264        ux = Ray.vector[0]
265        uy = Ray.vector[1]
266        uz = Ray.vector[2]
267        xA = Ray.point[0]
268        yA = Ray.point[1]
269        zA = Ray.point[2]
270    
271        da = ux**2 + uy**2
272        db = 2 * (ux*xA + uy*yA) - 2*self.p*uz
273        dc = xA**2 + yA**2 - 2*self.p*zA
274        
275        Solution = mgeo.SolverQuadratic(da,db,dc)
276        Solution = mgeo.KeepPositiveSolution(Solution)
277        
278        ListPointIntersection = []
279        for t in Solution:
280            I = Ray.vector*t + Ray.point
281            if self.support._IncludeSupport(I-self.get_centre()):
282                ListPointIntersection.append(I)
283
284        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
285    
286    def get_normal(self, Point):
287         """ Return the normal unit vector on the paraboloid surface at point Point """
288         Gradient = np.zeros(3)
289         Gradient[0] = Point[0]
290         Gradient[1] = Point[1]
291         Gradient[2] = -self.p
292         return mgeo.Normalize(Gradient)
293    
294    def get_centre(self):
295        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
296        return np.array([self.feff*np.sin(self.offaxisangle),0,self.p*0.5 - self.feff*np.cos(self.offaxisangle)])
297    
298    def get_grid3D(self,NbPoint):
299        """
300        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
301        sampling the support in a number NbPoints of points.
302        """
303        ListCoordXYZ = []
304        ListCoordXY = self.support._get_grid(NbPoint)
305        xc = self.feff*np.sin(self.offaxisangle)
306        for k in ListCoordXY:
307            z = ((k[0]+xc)**2 + k[1]**2)/2/self.p
308            ListCoordXYZ.append(np.array([k[0]+xc,k[1],z]))
309        return ListCoordXYZ

314class MirrorToroidal:
315    """
316    Toroidal mirror surface with eqn. $(\\sqrt{x**2 +z**2} - R)^2 + y^2 = r^2$,
317    where $R$ and $r$ the major and minor radii.
318    
319    ![Illustration of a toroidal mirror.](../documentation/toroidal.svg)
320    
321    Attributes
322    ----------
323        majorradius : float
324            Major radius of the toroid in mm.
325            
326        minorradius : float
327            Minor radius of the toroid in mm.
328  
329        support : Support-object from ModuleSupport
330        
331        type : str 'Toroidal Mirror'.
332            
333    Methods
334    ----------
335        get_normal(Point)
336        
337        get_centre()
338        
339        get_grid3D(NbPoints)
340        
341    """
342    def __init__(self, MajorRadius, MinorRadius, Support):
343        """
344        Parameters
345        ----------
346            MajorRadius : float
347                Major radius of the toroid in mm.
348            
349            MinorRadius : float
350                Minor radius of the toroid in mm.
351            
352            Support : Support-object from ModuleSupport
353
354        """
355        self.majorradius = MajorRadius
356        self.minorradius = MinorRadius
357        self.support = Support
358        self.type = 'Toroidal Mirror'   
359    
360    def _get_intersection(self, Ray):
361        """  Return the intersection point between Ray and the toroidal mirror surface. """
362        ux = Ray.vector[0]
363        uz = Ray.vector[2]
364        xA = Ray.point[0]
365        zA = Ray.point[2]
366        
367        G = 4.*self.majorradius**2 * (ux**2 + uz**2)
368        H = 8.*self.majorradius**2 * (ux*xA + uz*zA)
369        I = 4.*self.majorradius**2 * (xA**2 + zA**2)
370        J = np.dot(Ray.vector,Ray.vector)
371        K = 2.* np.dot(Ray.vector,Ray.point)
372        L = np.dot(Ray.point,Ray.point) + self.majorradius**2 - self.minorradius**2
373        
374        a = J**2
375        b = 2*J*K
376        c = 2*J*L + K**2 -G
377        d = 2*K*L - H
378        e = L**2 - I
379        
380        Solution = mgeo.SolverQuartic(a,b,c,d,e)
381        Solution = mgeo.KeepPositiveSolution(Solution)
382       
383        ListPointIntersection = []
384        for t in Solution:
385            I = Ray.vector*t + Ray.point
386            if I[2] < -self.majorradius and self.support._IncludeSupport(I): # For realistic mirror
387                ListPointIntersection.append(I)            
388        
389        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
390    
391    def get_normal(self, Point):
392        """ Return the normal unit vector on the toroidal surface at point Point. """
393        x = Point[0]
394        y = Point[1]
395        z = Point[2]
396        A = self.majorradius**2 - self.minorradius**2
397        
398        Gradient = np.zeros(3)
399        Gradient[0] = 4 * (x**3 + x*y**2 + x*z**2 + x*A) - 8*x*self.majorradius**2
400        Gradient[1] = 4 * (y**3 + y*x**2 + y*z**2 + y*A)
401        Gradient[2] = 4 * (z**3 + z*x**2 + z*y**2 + z*A) - 8*z*self.majorradius**2
402        
403        return mgeo.Normalize(Gradient)
404
405    def get_centre(self):
406        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
407        return np.array([0,0,-self.majorradius - self.minorradius])
408    
409    def get_grid3D(self,NbPoint):
410        """
411        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
412        sampling the support in a number NbPoints of points.
413        """
414        ListCoordXYZ = []
415        ListCoordXY = self.support._get_grid(NbPoint)
416        for k in ListCoordXY:
417            z = -np.sqrt( (np.sqrt(self.minorradius**2 - k[1]**2 ) + self.majorradius )**2 - k[0]**2)
418            ListCoordXYZ.append(np.array([k[0],k[1],z]))
419        return ListCoordXYZ

422def ReturnOptimalToroidalRadii(Focal : float, AngleIncidence : float) -> (float, float):
423    """
424    Useful helper function to get the optimal major and minor radii for a toroidal mirror to achieve a
425    focal length 'Focal' with an angle of incidence 'AngleIncidence' and with vanishing astigmatism.
426    
427    Parameters
428    ----------
429        Focal : float
430            Focal length in mm.
431            
432        AngleIncidence : int
433            Angle of incidence in degrees.
434    
435    Returns
436    -------
437        OptimalMajorRadius, OptimalMinorRadius : float, float.
438    """
439    AngleIncidenceRadian = AngleIncidence*np.pi/180
440    OptimalMajorRadius = 2*Focal*(1/np.cos(AngleIncidenceRadian) - np.cos(AngleIncidenceRadian))
441    OptimalMinorRadius = 2*Focal*np.cos(AngleIncidenceRadian)
442    return OptimalMajorRadius, OptimalMinorRadius

446class MirrorEllipsoidal:
447    """
448    Ellipdoidal mirror surface with eqn. $(x/a)^2 + (y/b)^2 + (z/b)^2 = 1$,
449    where $a$ and $b$ are semi major and semi minor axes.
450    
451    ![Illustration of a ellipsoidal mirror.](../documentation/ellipsoid.svg)
452
453    Attributes
454    ----------
455        a : float
456            Semi major axis of the ellipsoid in mm.
457            
458        b : float
459            Semi minor axis of the ellipsoid in mm.    
460
461        support : Support-object from ModuleSupport
462        
463        type : str 'Ellipsoidal Mirror'.
464             
465    Methods
466    ----------
467        get_normal(Point)
468        
469        get_centre()
470        
471        get_grid3D(NbPoints)
472         
473    """
474    def __init__(self, SemiMajorAxis, SemiMinorAxis, Support):
475        """
476        Parameters
477        ----------
478            SemiMajorAxis : float
479                Semi major axis of the ellipsoid in mm.
480            
481            SemiMinorAxis : float
482                Semi minor axis of the ellipsoid in mm.
483            
484            Support : Support-object from ModuleSupport
485
486        """
487        self.a = SemiMajorAxis
488        self.b = SemiMinorAxis
489        self.support = Support
490        self.type = 'Ellipsoidal Mirror'
491    
492    def _get_intersection(self, Ray):
493        """  Return the intersection point between Ray and the ellipsoidal mirror surface. """
494        ux = Ray.vector[0]
495        uy = Ray.vector[1]
496        uz = Ray.vector[2]
497        xA = Ray.point[0]
498        yA = Ray.point[1]
499        zA = Ray.point[2]
500        
501        da = (uy**2 + uz**2)/self.b**2 + (ux/self.a)**2
502        db = 2 * ( (uy*yA + uz*zA) /self.b**2 + (ux*xA)/self.a**2)
503        dc = (yA**2 + zA**2)/self.b**2 + (xA/self.a)**2 - 1
504    
505        Solution = mgeo.SolverQuadratic(da,db,dc)
506        Solution = mgeo.KeepPositiveSolution(Solution)
507        
508        ListPointIntersection = []
509        for t in Solution:
510            I = Ray.vector*t + Ray.point
511            if I[2] < 0 and self.support._IncludeSupport(I) :
512                ListPointIntersection.append(I)            
513                    
514        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
515    
516    def get_normal(self, Point):
517        """ Return the normal unit vector on the ellipsoidal surface at point Point """
518        Gradient = np.zeros(3)
519        
520        Gradient[0] = Point[0]/self.a**2
521        Gradient[1] = Point[1]/self.b**2
522        Gradient[2] = Point[2]/self.b**2
523        
524        return mgeo.Normalize(Gradient)
525
526    def get_centre(self):
527        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
528        return np.array([0,0,-self.b])
529    
530    def get_grid3D(self,NbPoint):
531        """
532        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
533        sampling the support in a number NbPoints of points.
534        """
535        ListCoordXYZ = []
536        ListCoordXY = self.support._get_grid(NbPoint)
537        for k in ListCoordXY:
538            z = -self.b*np.sqrt(1 - (k[0]/self.a)**2 - (k[1]/self.b)**2 )
539            ListCoordXYZ.append(np.array([k[0],k[1],z]))
540        
541        return ListCoordXYZ
542

657def _ReflectionMirrorRay(Mirror, PointMirror, Ray):
658    """
659    Returns the reflected ray according to the law of reflection.
660    Updates the reflected ray's incidence angle and path.
661    
662    Parameters
663    ----------
664        Mirror : Mirror-object
665        
666        PointMirror : np.ndarray
667            Point of reflection on the mirror surface.
668            
669        Ray : Ray-object
670
