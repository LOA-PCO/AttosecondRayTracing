ART.ModuleMirror

Provides classes for different mirror surfaces, which are types of optics. Think of these as the z-coordinates on top of the x-y-grid provided by the Support.

Also provides the function ReflectionMirrorRayList that returns the rays reflected on the mirror. Rays that do not hit the support are not further propagated.

 33class MirrorPlane:
 34    """
 35    A plane mirror.
 36    
 37    Attributes
 38    ----------
 39        support : [Support](ModuleSupport.html)-object from ModuleSupport
 40        
 41        type : 'Plane Mirror'.
 42            
 43    Methods
 44    ----------
 45        get_normal(Point)
 46        
 47        get_centre()
 48        
 49        get_grid3D(NbPoints)
 50    """
 51    def __init__(self, Support):
 52        """
 53        Parameters
 54        ----------
 55            Support : [Support](ModuleSupport.html)-object
 56
 57        """
 58        self.support = Support
 59        self.type = 'Plane Mirror'
 60
 61    def _get_intersection(self, Ray):
 62        """  Return the intersection point between Ray and the xy-plane  """
 63        t = -Ray.point[2] / Ray.vector[2]
 64        I = Ray.vector*t + Ray.point
 65        if t > 0 and self.support._IncludeSupport(I):
 66            PointIntersection = Ray.vector*t + Ray.point
 67        else: 
 68            PointIntersection = None
 69
 70        return PointIntersection
 71    
 72    
 73    def get_normal(self, Point : np.ndarray):
 74        """  Return normal unit vector in point 'Point' on the plane mirror. """
 75        Normal = np.array([0,0,1])
 76        return Normal
 77    
 78    def get_centre(self):
 79        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
 80        return np.array([0,0,0])
 81    
 82    def get_grid3D(self, NbPoints : int):
 83        """
 84        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
 85        sampling the support in a number NbPoints of points.
 86        """
 87        ListCoordXYZ = []
 88        ListCoordXY = self.support._get_grid(NbPoints)
 89        for k in ListCoordXY:
 90            z = 0
 91            ListCoordXYZ.append(np.array([k[0],k[1],z]))
 92        return ListCoordXYZ

 96class MirrorSpherical:
 97    """
 98    Spherical mirror surface with eqn. $x^2 + y^2 + z^2  = R^2$, where $R$ is the radius.
 99    
100    ![Illustration of a spherical mirror.](../documentation/sphericalmirror.svg)
101    
102    Attributes
103    ----------
104        radius : float
105            Radius of curvature. A postive value for concave mirror, a negative value for a convex mirror.
106    
107        support : [Support](ModuleSupport.html)-object
108        
109        type : str SphericalCC Mirror' or SphericalCX Mirror'
110            
111    Methods
112    ----------
113        get_normal(Point)
114        
115        get_centre()
116        
117        get_grid3D(NbPoints)
118    
119    """
120    def __init__(self, Radius, Support):
121        """
122        Parameters
123        ----------
124            Radius : float
125                The radius of curvature in mm. A postive value for concave mirror, a negative value for a convex mirror.
126            
127            Support : [Support](ModuleSupport.html)-object
128
129        """
130        if Radius <0:
131            self.type = 'SphericalCX Mirror'
132            self.radius = -Radius
133        else:
134            self.type = 'SphericalCC Mirror'
135            self.radius = Radius
136        
137        self.support = Support
138        
139    
140    def _get_intersection(self, Ray):
141        """  Return the intersection point between the ray and the sphere  """                
142        a = np.dot(Ray.vector,Ray.vector)
143        b = 2 * np.dot(Ray.vector, Ray.point)
144        c = np.dot(Ray.point, Ray.point) - self.radius**2
145        
146        Solution = mgeo.SolverQuadratic(a,b,c)
147        Solution = mgeo.KeepPositiveSolution(Solution)
148        
149        ListPointIntersection = []
150        for t in Solution:
151            I = Ray.vector*t + Ray.point
152            if I[2] < 0 and self.support._IncludeSupport(I):
153                ListPointIntersection.append(I) 
154                    
155        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
156    
157    def get_normal(self, Point):
158        """ Return the normal unit vector on the spherical surface at point Point """
159        Gradient = Point
160        return mgeo.Normalize(Gradient)
161    
162    def get_centre(self):
163        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
164        return np.array([0,0,-self.radius])
165    
166    def get_grid3D(self,NbPoint):
167        """
168        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
169        sampling the support in a number NbPoints of points.
170        """
171        ListCoordXYZ = []
172        ListCoordXY = self.support._get_grid(NbPoint)
173        for k in ListCoordXY:
174            z = -np.sqrt(self.radius**2 - (k[0]**2 + k[1]**2))
175            ListCoordXYZ.append(np.array([k[0],k[1],z]))
176        return ListCoordXYZ

180class MirrorParabolic:
181    """ 
182    A paraboloid with vertex at the origin [0,0,0] and symmetry axis z, 
183    with the center of the support shifted off axis along the x-direction.
184    Its eqn. is therefore $z = \\frac{1}{2p}[(x+x_c)^2 + y^2]$ where $p$ is the semi latus rectum,
185    equal to twice the focal lenght of the *mother* parabola (i.e. off-axis angle $\\alpha=0$).
186    
187    The effective focal length $f_\\mathrm{eff}$ is related to the semi latus rectum by 
188    $ p = f_\\mathrm{eff} (1+\\cos\\alpha)$, where $\\alpha$ is the off-axis angle.
189    
190    ![Illustration of a parabolic mirror.](../documentation/parabola.svg)
191
192    Attributes
193    ----------
194        offaxisangle : float
195            Off-axis angle of the parabola. Modifying this also updates p, keeping feff constant.
196            Attention: The off-axis angle must be *given in degrees*, but is stored and will be *returned in radian* !
197            
198        feff : float
199            Effective focal length of the parabola in mm. Modifying this also updates p, keeping offaxisangle constant.
200    
201        p : float
202            Semi latus rectum of the parabola in mm. Modifying this also updates feff, keeping offaxisangle constant.
203    
204        support : [Support](ModuleSupport.html)-object 
205        
206        type : str 'Parabolic Mirror'.
207            
208    Methods
209    ----------
210        get_normal(Point)
211        
212        get_centre()
213        
214        get_grid3D(NbPoints)
215    
216    """
217    def __init__(self, FocalEffective :float, OffAxisAngle :float, Support):
218        """
219        Parameters
220        ----------
221            FocalEffective : float
222                Effective focal length of the parabola in mm.
223            
224            OffAxisAngle : float
225                Off-axis angle *in degrees* of the parabola.
226            
227            Support : [Support](ModuleSupport.html)-object
228
229        """
230        self._offaxisangle = np.deg2rad(OffAxisAngle) 
231        self.support = Support
232        self.type = 'Parabolic Mirror'
233        self._feff = FocalEffective  #effective focal length
234        self._p = FocalEffective *(1+np.cos(self.offaxisangle)) #semi latus rectum
235    
236    @property
237    def offaxisangle(self): 
238        return self._offaxisangle
239       
240    @offaxisangle.setter 
241    def offaxisangle(self, OffAxisAngle): 
242        self._offaxisangle = np.deg2rad(OffAxisAngle)
243        self._p = self._feff *(1+np.cos(self._offaxisangle)) #make sure to always update p
244        
245    @property
246    def feff(self): 
247        return self._feff
248       
249    @feff.setter 
250    def feff(self, FocalEffective): 
251        self._feff = FocalEffective
252        self._p = self._feff *(1+np.cos(self.offaxisangle)) #make sure to always update p
253        
254    @property
255    def p(self): 
256        return self._p
257       
258    @p.setter 
259    def p(self, SemiLatusRectum): 
260        self._p = SemiLatusRectum
261        self._feff = self._p/(1+np.cos(self.offaxisangle))  #make sure to always update feff
262 
263    
264    def _get_intersection(self, Ray):
265        """  Return the intersection point between the ray and the parabola  """
266        ux = Ray.vector[0]
267        uy = Ray.vector[1]
268        uz = Ray.vector[2]
269        xA = Ray.point[0]
270        yA = Ray.point[1]
271        zA = Ray.point[2]
272    
273        da = ux**2 + uy**2
274        db = 2 * (ux*xA + uy*yA) - 2*self.p*uz
275        dc = xA**2 + yA**2 - 2*self.p*zA
276        
277        Solution = mgeo.SolverQuadratic(da,db,dc)
278        Solution = mgeo.KeepPositiveSolution(Solution)
279        
280        ListPointIntersection = []
281        for t in Solution:
282            I = Ray.vector*t + Ray.point
283            if self.support._IncludeSupport(I-self.get_centre()):
284                ListPointIntersection.append(I)
285
286        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
287    
288    def get_normal(self, Point):
289         """ Return the normal unit vector on the paraboloid surface at point Point """
290         Gradient = np.zeros(3)
291         Gradient[0] = Point[0]
292         Gradient[1] = Point[1]
293         Gradient[2] = -self.p
294         return mgeo.Normalize(Gradient)
295    
296    def get_centre(self):
297        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
298        return np.array([self.feff*np.sin(self.offaxisangle),0,self.p*0.5 - self.feff*np.cos(self.offaxisangle)])
299    
300    def get_grid3D(self,NbPoint):
301        """
302        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
303        sampling the support in a number NbPoints of points.
304        """
305        ListCoordXYZ = []
306        ListCoordXY = self.support._get_grid(NbPoint)
307        xc = self.feff*np.sin(self.offaxisangle)
308        for k in ListCoordXY:
309            z = ((k[0]+xc)**2 + k[1]**2)/2/self.p
310            ListCoordXYZ.append(np.array([k[0]+xc,k[1],z]))
311        return ListCoordXYZ

316class MirrorToroidal:
317    """
318    Toroidal mirror surface with eqn. $(\\sqrt{x**2 +z**2} - R)^2 + y^2 = r^2$,
319    where $R$ and $r$ the major and minor radii.
320    
321    ![Illustration of a toroidal mirror.](../documentation/toroidal.svg)
322    
323    Attributes
324    ----------
325        majorradius : float
326            Major radius of the toroid in mm.
327            
328        minorradius : float
329            Minor radius of the toroid in mm.
330  
331        support : [Support](ModuleSupport.html)-object 
332        
333        type : str 'Toroidal Mirror'.
334            
335    Methods
336    ----------
337        get_normal(Point)
338        
339        get_centre()
340        
341        get_grid3D(NbPoints)
342        
343    """
344    def __init__(self, MajorRadius, MinorRadius, Support):
345        """
346        Parameters
347        ----------
348            MajorRadius : float
349                Major radius of the toroid in mm.
350            
351            MinorRadius : float
352                Minor radius of the toroid in mm.
353            
354            Support : [Support](ModuleSupport.html)-object
355
356        """
357        self.majorradius = MajorRadius
358        self.minorradius = MinorRadius
359        self.support = Support
360        self.type = 'Toroidal Mirror'   
361    
362    def _get_intersection(self, Ray):
363        """  Return the intersection point between Ray and the toroidal mirror surface. """
364        ux = Ray.vector[0]
365        uz = Ray.vector[2]
366        xA = Ray.point[0]
367        zA = Ray.point[2]
368        
369        G = 4.*self.majorradius**2 * (ux**2 + uz**2)
370        H = 8.*self.majorradius**2 * (ux*xA + uz*zA)
371        I = 4.*self.majorradius**2 * (xA**2 + zA**2)
372        J = np.dot(Ray.vector,Ray.vector)
373        K = 2.* np.dot(Ray.vector,Ray.point)
374        L = np.dot(Ray.point,Ray.point) + self.majorradius**2 - self.minorradius**2
375        
376        a = J**2
377        b = 2*J*K
378        c = 2*J*L + K**2 -G
379        d = 2*K*L - H
380        e = L**2 - I
381        
382        Solution = mgeo.SolverQuartic(a,b,c,d,e)
383        Solution = mgeo.KeepPositiveSolution(Solution)
384       
385        ListPointIntersection = []
386        for t in Solution:
387            I = Ray.vector*t + Ray.point
388            if I[2] < -self.majorradius and self.support._IncludeSupport(I): # For realistic mirror
389                ListPointIntersection.append(I)            
390        
391        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
392    
393    def get_normal(self, Point):
394        """ Return the normal unit vector on the toroidal surface at point Point. """
395        x = Point[0]
396        y = Point[1]
397        z = Point[2]
398        A = self.majorradius**2 - self.minorradius**2
399        
400        Gradient = np.zeros(3)
401        Gradient[0] = 4 * (x**3 + x*y**2 + x*z**2 + x*A) - 8*x*self.majorradius**2
402        Gradient[1] = 4 * (y**3 + y*x**2 + y*z**2 + y*A)
403        Gradient[2] = 4 * (z**3 + z*x**2 + z*y**2 + z*A) - 8*z*self.majorradius**2
404        
405        return mgeo.Normalize(Gradient)
406
407    def get_centre(self):
408        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
409        return np.array([0,0,-self.majorradius - self.minorradius])
410    
411    def get_grid3D(self,NbPoint):
412        """
413        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
414        sampling the support in a number NbPoints of points.
415        """
416        ListCoordXYZ = []
417        ListCoordXY = self.support._get_grid(NbPoint)
418        for k in ListCoordXY:
419            z = -np.sqrt( (np.sqrt(self.minorradius**2 - k[1]**2 ) + self.majorradius )**2 - k[0]**2)
420            ListCoordXYZ.append(np.array([k[0],k[1],z]))
421        return ListCoordXYZ

424def ReturnOptimalToroidalRadii(Focal : float, AngleIncidence : float) -> (float, float):
425    """
426    Useful helper function to get the optimal major and minor radii for a toroidal mirror to achieve a
427    focal length 'Focal' with an angle of incidence 'AngleIncidence' and with vanishing astigmatism.
428    
429    Parameters
430    ----------
431        Focal : float
432            Focal length in mm.
433            
434        AngleIncidence : int
435            Angle of incidence in degrees.
436    
437    Returns
438    -------
439        OptimalMajorRadius, OptimalMinorRadius : float, float.
440    """
441    AngleIncidenceRadian = AngleIncidence*np.pi/180
442    OptimalMajorRadius = 2*Focal*(1/np.cos(AngleIncidenceRadian) - np.cos(AngleIncidenceRadian))
443    OptimalMinorRadius = 2*Focal*np.cos(AngleIncidenceRadian)
444    return OptimalMajorRadius, OptimalMinorRadius

448class MirrorEllipsoidal:
449    """
450    Ellipdoidal mirror surface with eqn. $(x/a)^2 + (y/b)^2 + (z/b)^2 = 1$,
451    where $a$ and $b$ are semi major and semi minor axes.
452    
453    ![Illustration of a ellipsoidal mirror.](../documentation/ellipsoid.svg)
454
455    Attributes
456    ----------
457        a : float
458            Semi major axis of the ellipsoid in mm.
459            
460        b : float
461            Semi minor axis of the ellipsoid in mm.    
462
463        support : [Support](ModuleSupport.html)-object
464        
465        type : str 'Ellipsoidal Mirror'.
466             
467    Methods
468    ----------
469        get_normal(Point)
470        
471        get_centre()
472        
473        get_grid3D(NbPoints)
474         
475    """
476    def __init__(self, SemiMajorAxis, SemiMinorAxis, Support):
477        """
478        Parameters
479        ----------
480            SemiMajorAxis : float
481                Semi major axis of the ellipsoid in mm.
482            
483            SemiMinorAxis : float
484                Semi minor axis of the ellipsoid in mm.
485            
486            Support : [Support](ModuleSupport.html)-object
487
488        """
489        self.a = SemiMajorAxis
490        self.b = SemiMinorAxis
491        self.support = Support
492        self.type = 'Ellipsoidal Mirror'
493    
494    def _get_intersection(self, Ray):
495        """  Return the intersection point between Ray and the ellipsoidal mirror surface. """
496        ux = Ray.vector[0]
497        uy = Ray.vector[1]
498        uz = Ray.vector[2]
499        xA = Ray.point[0]
500        yA = Ray.point[1]
501        zA = Ray.point[2]
502        
503        da = (uy**2 + uz**2)/self.b**2 + (ux/self.a)**2
504        db = 2 * ( (uy*yA + uz*zA) /self.b**2 + (ux*xA)/self.a**2)
505        dc = (yA**2 + zA**2)/self.b**2 + (xA/self.a)**2 - 1
506    
507        Solution = mgeo.SolverQuadratic(da,db,dc)
508        Solution = mgeo.KeepPositiveSolution(Solution)
509        
510        ListPointIntersection = []
511        for t in Solution:
512            I = Ray.vector*t + Ray.point
513            if I[2] < 0 and self.support._IncludeSupport(I) :
514                ListPointIntersection.append(I)            
515                    
516        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
517    
518    def get_normal(self, Point):
519        """ Return the normal unit vector on the ellipsoidal surface at point Point """
520        Gradient = np.zeros(3)
521        
522        Gradient[0] = Point[0]/self.a**2
523        Gradient[1] = Point[1]/self.b**2
524        Gradient[2] = Point[2]/self.b**2
525        
526        return mgeo.Normalize(Gradient)
527
528    def get_centre(self):
529        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
530        return np.array([0,0,-self.b])
531    
532    def get_grid3D(self,NbPoint):
533        """
534        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
535        sampling the support in a number NbPoints of points.
536        """
537        ListCoordXYZ = []
538        ListCoordXY = self.support._get_grid(NbPoint)
539        for k in ListCoordXY:
540            z = -self.b*np.sqrt(1 - (k[0]/self.a)**2 - (k[1]/self.b)**2 )
541            ListCoordXYZ.append(np.array([k[0],k[1],z]))
542        
543        return ListCoordXYZ

659def _ReflectionMirrorRay(Mirror, PointMirror, Ray):
660    """
661    Returns the reflected ray according to the law of reflection.
662    Updates the reflected ray's incidence angle and path.
663    
664    Parameters
665    ----------
