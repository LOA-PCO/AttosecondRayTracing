ART.ModuleSupport

Provides classes for different shapes of a support for optics.

A support fixes the spatial extent and outer shape of optics - think of it as a substrate. Technically, the support is a section of the x-y-plane in the element coordinate frame.

The support will become an attribute of Mirror or Mask objects.

 22class SupportRound:
 23    """
 24    A round support for optics.
 25
 26    Attributes
 27    ----------
 28        radius : float
 29            The radius of the support in mm.
 30            
 31    """
 32    def __init__(self, Radius : float):
 33        """
 34        Parameters
 35        ----------
 36            Radius : float
 37                The radius of the support in mm.
 38            
 39        """
 40        self.radius = Radius
 41        
 42    def _IncludeSupport(self, Point):
 43        """ Returns whether 2D-Point is within the rectangular Support. """
 44        return mgeo.IncludeDisk(self.radius, Point)
 45    
 46    def _get_grid(self, NbPoint):
 47        """
 48        Returns a list of 2D-numpy-arrays with the coordinates a number NbPoints of points,
 49        distributed as a Vogel-spiral on the support, with the origin in the center of the support.
 50        """
 51        MatrixXY = mgeo.SpiralVogel(NbPoint, self.radius)
 52        ListCoordXY = []
 53        for k in range(NbPoint):
 54            x = MatrixXY[k,0]
 55            y = MatrixXY[k,1]
 56            ListCoordXY.append(np.array([x,y]))
 57        return ListCoordXY
 58    
 59    def _ContourSupport(self, Figure):
 60        """ Draws support contour in MirrorProjection plots. """
 61        axe = Figure.add_subplot(111, aspect='equal')
 62        axe.add_patch(patches.Circle((0,0), self.radius,alpha=0.08))
 63        return axe

 67class SupportRoundHole:
 68    """
 69    A round support for optics with a round hole.
 70    
 71    Attributes
 72    ----------
 73        radius : float
 74            The radius of the support in mm.
 75        
 76        radiushole : float
 77            The radius of the hole in mm.
 78             
 79        centerholeX : float
 80            The x-cordinate of the hole's centre, in mm.
 81             
 82        centerholeY : float
 83            The y-cordinate of the hole's centre, in mm.
 84            
 85    """
 86    def __init__(self, Radius, RadiusHole, CenterHoleX, CenterHoleY):
 87        """
 88        Parameters
 89        ----------
 90        Radius : float
 91            The radius of the support in mm.
 92            
 93        RadiusHole : float
 94            The radius of the hole in mm.
 95             
 96        CenterHoleX : float
 97            The x-cordinate of the hole's centre, in mm.
 98             
 99        CenterHoleY : float
100            The y-cordinate of the hole's centre, in mm.
101            
102        """
103        self.radius = Radius
104        self.radiushole = RadiusHole
105        self.centerholeX = CenterHoleX
106        self.centerholeY = CenterHoleY
107        
108    def _IncludeSupport(self, Point):
109        """ Returns whether 2D-Point is within the rectangular Support. """
110        return mgeo.IncludeDisk(self.radius, Point) and not(mgeo.IncludeDisk(self.radiushole, Point-np.array([self.centerholeX,self.centerholeY,0])))
111    
112    def _get_grid(self,NbPoint):
113        """
114        Returns a list of 2D-numpy-arrays with the coordinates a number NbPoints of points,
115        distributed as a Vogel-spiral on the support, with the origin in the center of the support.
116        """
117        MatrixXY = mgeo.SpiralVogel(NbPoint, self.radius)       
118        ListCoordXY = []
119        for k in range(NbPoint):
120            x = MatrixXY[k,0]
121            y = MatrixXY[k,1]
122            if (x-self.centerholeX)**2 + (y-self.centerholeY)**2 > self.radiushole**2:
123                ListCoordXY.append(np.array([x,y]))
124        return ListCoordXY
125    
126    def _ContourSupport(self, Figure):
127        """ Draws support contour in MirrorProjection plots. """
128        axe = Figure.add_subplot(111, aspect='equal')
129        axe.add_patch(patches.Circle((0,0), self.radius,alpha=0.08))
130        axe.add_patch(patches.Circle((self.centerholeX,self.centerholeY), self.radiushole, color='white',alpha=1))
131        return axe

134class SupportRectangle:
135    """
136    A rectangular support for optics.
137    
138    Attributes
139    ----------
140        dimX : float
141            The dimension in mm along x.
142        
143        dimY : float
144            The dimension in mm along y.   
145            
146    """
147    def __init__(self, DimensionX : float, DimensionY : float):
148        """
149        Parameters
150        ----------
151        DimensionX : float
152            The dimension in mm along x.
153            
154        DimensionY : float
155            The dimension in mm along y.
156            
157        """
158        self.dimX = DimensionX
159        self.dimY = DimensionY
160        
161    def _IncludeSupport(self, Point : np.ndarray) -> bool:
162        """ Returns whether 2D-Point is within the rectangular Support. """
163        return mgeo.IncludeRectangle(self.dimX, self.dimY, Point)
164    
165    def _get_grid(self, NbPoints : int) -> list[np.ndarray]:
166        """
167        Returns a list of 2D-numpy-arrays with the coordinates a number NbPoints of points,
168        distributed as a regular grid on the support, with the origin in the center of the support.
169        """
170        nbx = int(np.sqrt(self.dimX/self.dimY*NbPoints + 0.25*(self.dimX-self.dimY)**2 / self.dimY**2) - 0.5 * (self.dimX-self.dimY)/self.dimY)
171        nby = int(NbPoints/nbx) 
172        x = np.linspace(-self.dimX/2,self.dimX/2,nbx)
173        y = np.linspace(-self.dimY/2,self.dimY/2,nby)    
174        ListCoordXY = []
175        for i in x:
176            for j in y:
177                 ListCoordXY.append(np.array([i,j]))
178        return ListCoordXY
179
180    def _ContourSupport(self, Figure):
181        """ Draws support contour in MirrorProjection plots. """
182        axe = Figure.add_subplot(111, aspect='equal')
183        axe.add_patch(patches.Rectangle((-self.dimX*0.5,-self.dimY*0.5), self.dimX, self.dimY,alpha=0.08))
184        return axe
