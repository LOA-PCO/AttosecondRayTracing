ART.ModuleMirror

Provides classes for different mirror surfaces, which are types of optics. Think of these as the z-coordinates on top of the x-y-grid provided by the Support.

Also provides the function ReflectionMirrorRayList that returns the rays reflected on the mirror. Rays that do not hit the support are not further propagated.

 41        type : 'Plane Mirror'.
 42            
 43    Methods

 47        get_centre()
 48        
 49        get_grid3D(NbPoints)
 50    """

 52        """
 53        Parameters
 54        ----------
 55            Support : [Support](ModuleSupport.html)-object
 56
 57        """
 58        self.support = Support
 59        self.type = 'Plane Mirror'
 60
 61    def _get_intersection(self, Ray):
 62        """  Return the intersection point between Ray and the xy-plane  """
 63        t = -Ray.point[2] / Ray.vector[2]
 64        I = Ray.vector*t + Ray.point
 65        if t > 0 and self.support._IncludeSupport(I):
 66            PointIntersection = Ray.vector*t + Ray.point
 67        else: 
 68            PointIntersection = None
 69
 70        return PointIntersection
 71

 74        """  Return normal unit vector in point 'Point' on the plane mirror. """

 76        return Normal

 79        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """

 83        """

 85        sampling the support in a number NbPoints of points.
 86        """
 87        ListCoordXYZ = []
 88        ListCoordXY = self.support._get_grid(NbPoints)
 89        for k in ListCoordXY:
 90            z = 0
 91            ListCoordXYZ.append(np.array([k[0],k[1],z]))
 92        return ListCoordXYZ

 96class MirrorSpherical:
 97    """
 98    Spherical mirror surface with eqn. $x^2 + y^2 + z^2  = R^2$, where $R$ is the radius.
 99    
100    ![Illustration of a spherical mirror.](../docs/sphericalmirror.svg)
101    
102    Attributes
103    ----------
104        radius : float
105            Radius of curvature. A postive value for concave mirror, a negative value for a convex mirror.
106    
107        support : [Support](ModuleSupport.html)-object
108        
109        type : str SphericalCC Mirror' or SphericalCX Mirror'
110            
111    Methods
112    ----------
113        get_normal(Point)
114        
115        get_centre()
116        
117        get_grid3D(NbPoints)
118    
119    """
120    def __init__(self, Radius, Support):
121        """
122        Parameters
123        ----------
124            Radius : float
125                The radius of curvature in mm. A postive value for concave mirror, a negative value for a convex mirror.
126            
127            Support : [Support](ModuleSupport.html)-object
128
129        """
130        if Radius <0:
131            self.type = 'SphericalCX Mirror'
132            self.radius = -Radius
133        else:
134            self.type = 'SphericalCC Mirror'
135            self.radius = Radius
136        
137        self.support = Support
138        
139    
140    def _get_intersection(self, Ray):
141        """  Return the intersection point between the ray and the sphere  """                
142        a = np.dot(Ray.vector,Ray.vector)
143        b = 2 * np.dot(Ray.vector, Ray.point)
144        c = np.dot(Ray.point, Ray.point) - self.radius**2
145        
146        Solution = mgeo.SolverQuadratic(a,b,c)
147        Solution = mgeo.KeepPositiveSolution(Solution)
148        
149        ListPointIntersection = []
150        for t in Solution:
151            I = Ray.vector*t + Ray.point
152            if I[2] < 0 and self.support._IncludeSupport(I):
153                ListPointIntersection.append(I) 
154                    
155        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
156    
157    def get_normal(self, Point):
158        """ Return the normal unit vector on the spherical surface at point Point """
159        Gradient = Point
160        return mgeo.Normalize(Gradient)
161    
162    def get_centre(self):
163        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
164        return np.array([0,0,-self.radius])
165    
166    def get_grid3D(self,NbPoint):
167        """
168        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
169        sampling the support in a number NbPoints of points.
170        """
171        ListCoordXYZ = []
172        ListCoordXY = self.support._get_grid(NbPoint)
173        for k in ListCoordXY:
174            z = -np.sqrt(self.radius**2 - (k[0]**2 + k[1]**2))
175            ListCoordXYZ.append(np.array([k[0],k[1],z]))
176        return ListCoordXYZ

180class MirrorParabolic:
181    """ 
182    A paraboloid with vertex at the origin $O=[0,0,0]$ and symmetry axis z. 
183    Its eqn. is therefore $z = \\frac{1}{4f}[(x)^2 + y^2]$ where $f$ is the focal
184    lenght of the *mother* parabola (i.e. measured from its center at $O$ to the focal point $F$).
185    
186    The center of the support is shifted along the x-direction by the off-axis distance $x_c$.
187    This leads to an *effective focal length* $f_\\mathrm{eff}$, measured from the shifted center
188    of the support  $P$ to the focal point $F$.
189    It is related to the mother focal length by $f = f_\\mathrm{eff} \\cos^2(\\alpha/2) $,
190    or equivalently $ p = 2f = f_\\mathrm{eff} (1+\\cos\\alpha)$, where $\\alpha$
191    is the off-axis angle, and $p = 2f$ is called the semi latus rectum.
192    
193    Another useful relationship is that between the off-axis distance and the resulting
194    off-axis angle: $x_c = 2 f \\tan(\\alpha/2)$.
195    
196    
197    ![Illustration of a parabolic mirror.](../docs/parabola.svg)
198
199    Attributes
200    ----------
201        offaxisangle : float
202            Off-axis angle of the parabola. Modifying this also updates p, keeping feff constant.
203            Attention: The off-axis angle must be *given in degrees*, but is stored and will be *returned in radian* !
204            
205        feff : float
206            Effective focal length of the parabola in mm. Modifying this also updates p, keeping offaxisangle constant.
207    
208        p : float
209            Semi latus rectum of the parabola in mm. Modifying this also updates feff, keeping offaxisangle constant.
210    
211        support : [Support](ModuleSupport.html)-object 
212        
213        type : str 'Parabolic Mirror'.
214            
215    Methods
216    ----------
217        get_normal(Point)
218        
219        get_centre()
220        
221        get_grid3D(NbPoints)
222    
223    """
224    def __init__(self, FocalEffective :float, OffAxisAngle :float, Support):
225        """
226        Parameters
227        ----------
228            FocalEffective : float
229                Effective focal length of the parabola in mm.
230            
231            OffAxisAngle : float
232                Off-axis angle *in degrees* of the parabola.
233            
234            Support : [Support](ModuleSupport.html)-object
235
236        """
237        self._offaxisangle = np.deg2rad(OffAxisAngle) 
238        self.support = Support
239        self.type = 'Parabolic Mirror'
240        self._feff = FocalEffective  #effective focal length
241        self._p = FocalEffective *(1+np.cos(self.offaxisangle)) #semi latus rectum, =2*focal length of mother parabola
242    
243    @property
244    def offaxisangle(self): 
245        return self._offaxisangle
246       
247    @offaxisangle.setter 
248    def offaxisangle(self, OffAxisAngle): 
249        self._offaxisangle = np.deg2rad(OffAxisAngle) #store (and later return) in radians!
250        self._p = self._feff *(1+np.cos(self._offaxisangle)) #make sure to always update p
251        
252    @property
253    def feff(self): 
254        return self._feff
255       
256    @feff.setter 
257    def feff(self, FocalEffective): 
258        self._feff = FocalEffective
259        self._p = self._feff *(1+np.cos(self.offaxisangle)) #make sure to always update p
260        
261    @property
262    def p(self): 
263        return self._p
264       
265    @p.setter 
266    def p(self, SemiLatusRectum): 
267        self._p = SemiLatusRectum
268        self._feff = self._p/(1+np.cos(self.offaxisangle))  #make sure to always update feff
269 
270    
271    def _get_intersection(self, Ray):
272        """  Return the intersection point between the ray and the parabola  """
273        ux = Ray.vector[0]
274        uy = Ray.vector[1]
275        uz = Ray.vector[2]
276        xA = Ray.point[0]
277        yA = Ray.point[1]
278        zA = Ray.point[2]
279    
280        da = ux**2 + uy**2
281        db = 2 * (ux*xA + uy*yA) - 2*self._p*uz
282        dc = xA**2 + yA**2 - 2*self._p*zA
283        
284        Solution = mgeo.SolverQuadratic(da,db,dc)
285        Solution = mgeo.KeepPositiveSolution(Solution)
286        
287        ListPointIntersection = []
288        for t in Solution:
289            I = Ray.vector*t + Ray.point
290            if self.support._IncludeSupport(I-self.get_centre()):
291                ListPointIntersection.append(I)
292
293        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
294    
295    def get_normal(self, Point):
296         """ Return the normal unit vector on the paraboloid surface at point Point """
297         Gradient = np.zeros(3)
298         Gradient[0] = Point[0]
299         Gradient[1] = Point[1]
300         Gradient[2] = -self._p
301         return mgeo.Normalize(Gradient)
302    
303    def get_centre(self):
304        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
305        return np.array([self.feff*np.sin(self.offaxisangle),0,self._p*0.5 - self.feff*np.cos(self.offaxisangle)])
306    
307    def get_grid3D(self,NbPoint):
308        """
309        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
310        sampling the support in a number NbPoints of points.
311        """
312        ListCoordXYZ = []
313        ListCoordXY = self.support._get_grid(NbPoint)
314        xc = self.feff*np.sin(self.offaxisangle)
315        for k in ListCoordXY:
316            z = ((k[0]+xc)**2 + k[1]**2)/2/self._p
317            ListCoordXYZ.append(np.array([k[0]+xc,k[1],z]))
318        return ListCoordXYZ

323class MirrorToroidal:
324    """
325    Toroidal mirror surface with eqn. $(\\sqrt{x**2 +z**2} - R)^2 + y^2 = r^2$,
326    where $R$ and $r$ the major and minor radii.
327    
328    ![Illustration of a toroidal mirror.](../docs/toroidal.svg)
329    
330    Attributes
331    ----------
332        majorradius : float
333            Major radius of the toroid in mm.
334            
335        minorradius : float
336            Minor radius of the toroid in mm.
337  
338        support : [Support](ModuleSupport.html)-object 
339        
340        type : str 'Toroidal Mirror'.
341            
342    Methods
343    ----------
344        get_normal(Point)
345        
346        get_centre()
347        
348        get_grid3D(NbPoints)
349        
350    """
351    def __init__(self, MajorRadius, MinorRadius, Support):
352        """
353        Parameters
354        ----------
355            MajorRadius : float
356                Major radius of the toroid in mm.
357            
358            MinorRadius : float
359                Minor radius of the toroid in mm.
360            
361            Support : [Support](ModuleSupport.html)-object
362
363        """
364        self.majorradius = MajorRadius
365        self.minorradius = MinorRadius
366        self.support = Support
367        self.type = 'Toroidal Mirror'   
368    
369    def _get_intersection(self, Ray):
370        """  Return the intersection point between Ray and the toroidal mirror surface. """
371        ux = Ray.vector[0]
372        uz = Ray.vector[2]
373        xA = Ray.point[0]
374        zA = Ray.point[2]
375        
376        G = 4.*self.majorradius**2 * (ux**2 + uz**2)
377        H = 8.*self.majorradius**2 * (ux*xA + uz*zA)
378        I = 4.*self.majorradius**2 * (xA**2 + zA**2)
379        J = np.dot(Ray.vector,Ray.vector)
380        K = 2.* np.dot(Ray.vector,Ray.point)
381        L = np.dot(Ray.point,Ray.point) + self.majorradius**2 - self.minorradius**2
382        
383        a = J**2
384        b = 2*J*K
385        c = 2*J*L + K**2 -G
386        d = 2*K*L - H
387        e = L**2 - I
388        
389        Solution = mgeo.SolverQuartic(a,b,c,d,e)
390        Solution = mgeo.KeepPositiveSolution(Solution)
391       
392        ListPointIntersection = []
393        for t in Solution:
394            I = Ray.vector*t + Ray.point
395            if I[2] < -self.majorradius and self.support._IncludeSupport(I): # For realistic mirror
396                ListPointIntersection.append(I)            
397        
398        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
399    
400    def get_normal(self, Point):
401        """ Return the normal unit vector on the toroidal surface at point Point. """
402        x = Point[0]
403        y = Point[1]
404        z = Point[2]
405        A = self.majorradius**2 - self.minorradius**2
406        
407        Gradient = np.zeros(3)
408        Gradient[0] = 4 * (x**3 + x*y**2 + x*z**2 + x*A) - 8*x*self.majorradius**2
409        Gradient[1] = 4 * (y**3 + y*x**2 + y*z**2 + y*A)
410        Gradient[2] = 4 * (z**3 + z*x**2 + z*y**2 + z*A) - 8*z*self.majorradius**2
411        
412        return mgeo.Normalize(Gradient)
413
414    def get_centre(self):
415        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
416        return np.array([0,0,-self.majorradius - self.minorradius])
417    
418    def get_grid3D(self,NbPoint):
419        """
420        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
421        sampling the support in a number NbPoints of points.
422        """
423        ListCoordXYZ = []
424        ListCoordXY = self.support._get_grid(NbPoint)
425        for k in ListCoordXY:
426            z = -np.sqrt( (np.sqrt(self.minorradius**2 - k[1]**2 ) + self.majorradius )**2 - k[0]**2)
427            ListCoordXYZ.append(np.array([k[0],k[1],z]))
428        return ListCoordXYZ

431def ReturnOptimalToroidalRadii(Focal : float, AngleIncidence : float) -> (float, float):
432    """
433    Useful helper function to get the optimal major and minor radii for a toroidal mirror to achieve a
434    focal length 'Focal' with an angle of incidence 'AngleIncidence' and with vanishing astigmatism.
435    
436    Parameters
437    ----------
438        Focal : float
439            Focal length in mm.
440            
441        AngleIncidence : int
442            Angle of incidence in degrees.
443    
444    Returns
445    -------
446        OptimalMajorRadius, OptimalMinorRadius : float, float.
447    """
448    AngleIncidenceRadian = AngleIncidence*np.pi/180
449    OptimalMajorRadius = 2*Focal*(1/np.cos(AngleIncidenceRadian) - np.cos(AngleIncidenceRadian))
450    OptimalMinorRadius = 2*Focal*np.cos(AngleIncidenceRadian)
451    return OptimalMajorRadius, OptimalMinorRadius

455class MirrorEllipsoidal:
456    """
457    Ellipdoidal mirror surface with eqn. $(x/a)^2 + (y/b)^2 + (z/b)^2 = 1$,
458    where $a$ and $b$ are semi major and semi minor axes.
459    
460    ![Illustration of a ellipsoidal mirror.](../docs/ellipsoid.svg)
461
462    Attributes
463    ----------
464        a : float
465            Semi major axis of the ellipsoid in mm.
466            
467        b : float
468            Semi minor axis of the ellipsoid in mm.    
469
470        support : [Support](ModuleSupport.html)-object
471        
472        type : str 'Ellipsoidal Mirror'.
473             
474    Methods
475    ----------
476        get_normal(Point)
477        
478        get_centre()
479        
480        get_grid3D(NbPoints)
481         
482    """
483    def __init__(self, SemiMajorAxis, SemiMinorAxis, Support):
484        """
485        Parameters
486        ----------
487            SemiMajorAxis : float
488                Semi major axis of the ellipsoid in mm.
489            
490            SemiMinorAxis : float
491                Semi minor axis of the ellipsoid in mm.
492            
493            Support : [Support](ModuleSupport.html)-object
494
495        """
496        self.a = SemiMajorAxis
497        self.b = SemiMinorAxis
498        self.support = Support
499        self.type = 'Ellipsoidal Mirror'
500    
501    def _get_intersection(self, Ray):
502        """  Return the intersection point between Ray and the ellipsoidal mirror surface. """
503        ux = Ray.vector[0]
504        uy = Ray.vector[1]
505        uz = Ray.vector[2]
506        xA = Ray.point[0]
507        yA = Ray.point[1]
508        zA = Ray.point[2]
509        
510        da = (uy**2 + uz**2)/self.b**2 + (ux/self.a)**2
511        db = 2 * ( (uy*yA + uz*zA) /self.b**2 + (ux*xA)/self.a**2)
512        dc = (yA**2 + zA**2)/self.b**2 + (xA/self.a)**2 - 1
513    
514        Solution = mgeo.SolverQuadratic(da,db,dc)
515        Solution = mgeo.KeepPositiveSolution(Solution)
516        
517        ListPointIntersection = []
518        for t in Solution:
519            I = Ray.vector*t + Ray.point
520            if I[2] < 0 and self.support._IncludeSupport(I) :
521                ListPointIntersection.append(I)            
522                    
523        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
524    
525    def get_normal(self, Point):
526        """ Return the normal unit vector on the ellipsoidal surface at point Point """
527        Gradient = np.zeros(3)
528        
529        Gradient[0] = Point[0]/self.a**2
530        Gradient[1] = Point[1]/self.b**2
531        Gradient[2] = Point[2]/self.b**2
532        
533        return mgeo.Normalize(Gradient)
534
535    def get_centre(self):
536        """  Return 3D coordinates of the point on the mirror surface at the center of its support. """
537        return np.array([0,0,-self.b])
538    
539    def get_grid3D(self,NbPoint):
540        """
541        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
542        sampling the support in a number NbPoints of points.
543        """
544        ListCoordXYZ = []
545        ListCoordXY = self.support._get_grid(NbPoint)
546        for k in ListCoordXY:
547            z = -self.b*np.sqrt(1 - (k[0]/self.a)**2 - (k[1]/self.b)**2 )
548            ListCoordXYZ.append(np.array([k[0],k[1],z]))
549        
550        return ListCoordXYZ
