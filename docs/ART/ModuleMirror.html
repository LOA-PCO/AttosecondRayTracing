ART.ModuleMirror

Provides classes for different mirror surfaces, which are types of optics.

Think of these as the z-coordinates on top of the x-y-grid provided by the ART.ModuleSupport.Support.

Also provides the function ReflectionMirrorRayList that returns the rays reflected on the mirror. Rays that do not hit the support are not propagated any further.

 27def _IntersectionRayMirror(PointMirror, ListPointIntersectionMirror):
 28    """When the ray has pierced the mirror twice, select the first point, otherwise just keep that one."""
 29    if len(ListPointIntersectionMirror) == 2:

 38class MirrorPlane:
 39    """
 40    A plane mirror.
 41
 42    Attributes
 43    ----------
 44        support : ART.ModuleSupport.Support
 45            Mirror support
 46        type : str = 'Plane Mirror'
 47            Human readable mirror type
 48    Methods
 49    -------
 50        MirrorPlane.get_normal(Point)
 51
 52        MirrorPlane.get_centre()
 53
 54        MirrorPlane.get_grid3D(NbPoints)
 55    """
 56
 57    def __init__(self, Support):
 58        """
 59        Create a plane mirror on a given Support.
 60
 61        Parameters
 62        ----------
 63            Support : ART.ModuleSupport.Support
 64
 65        """
 66        self.support = Support
 67        self.type = "Plane Mirror"
 68
 69    def _get_intersection(self, Ray):
 70        """Return the intersection point between Ray and the xy-plane."""
 71        t = -Ray.point[2] / Ray.vector[2]
 72        Intersect = Ray.vector * t + Ray.point
 73        if t > 0 and self.support._IncludeSupport(Intersect):
 74            PointIntersection = Ray.vector * t + Ray.point
 75        else:
 76            PointIntersection = None
 77
 78        return PointIntersection
 79
 80    def get_normal(self, Point: np.ndarray):
 81        """Return normal unit vector in point 'Point' on the plane mirror."""
 82        Normal = np.array([0, 0, 1])
 83        return Normal
 84
 85    def get_centre(self):
 86        """Return 3D coordinates of the point on the mirror surface at the center of its support."""
 87        return np.array([0, 0, 0])
 88
 89    def get_grid3D(self, NbPoint: int, **kwargs):
 90        """
 91        Get grid of points on mirror surface.
 92
 93        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
 94        sampling the support in a number NbPoints of points.
 95        """
 96        E = "edges" in kwargs and kwargs["edges"]
 97        ListCoordXYZ = []
 98        contour = int(round(0.1 * NbPoint))
 99        contours = self.support._Contour_points(contour, **kwargs)
100        if E:
101            contours, contour_edges = contours
102        ListCoordXY = contours + self.support._get_grid(NbPoint - contour)
103        for k in ListCoordXY:
104            z = 0
105            ListCoordXYZ.append(np.array([k[0], k[1], z]))
106        if E:
107            return ListCoordXYZ, contour_edges
108
109        return ListCoordXYZ

113class MirrorSpherical:
114    """
115    Spherical mirror surface with eqn. $x^2 + y^2 + z^2  = R^2$, where $R$ is the radius.
116
117    ![Illustration of a spherical mirror.](sphericalmirror.svg)
118
119    Attributes
120    ----------
121        radius : float
122            Radius of curvature. A postive value for concave mirror, a negative value for a convex mirror.
123
124        support : ART.ModuleSupport.Support
125
126        type : str SphericalCC Mirror' or SphericalCX Mirror'
127
128    Methods
129    -------
130        MirrorSpherical.get_normal(Point)
131
132        MirrorSpherical.get_centre()
133
134        MirrorSpherical.get_grid3D(NbPoints)
135
136    """
137
138    def __init__(self, Radius, Support):
139        """
140        Construct a spherical mirror.
141
142        Parameters
143        ----------
144            Radius : float
145                The radius of curvature in mm. A postive value for concave mirror, a negative value for a convex mirror.
146
147            Support : ART.ModuleSupport.Support
148
149        """
150        if Radius < 0:
151            self.type = "SphericalCX Mirror"
152            self.radius = -Radius
153        else:
154            self.type = "SphericalCC Mirror"
155            self.radius = Radius
156
157        self.support = Support
158
159    def _get_intersection(self, Ray):
160        """Return the intersection point between the ray and the sphere."""
161        a = np.dot(Ray.vector, Ray.vector)
162        b = 2 * np.dot(Ray.vector, Ray.point)
163        c = np.dot(Ray.point, Ray.point) - self.radius**2
164
165        Solution = mgeo.SolverQuadratic(a, b, c)
166        Solution = mgeo.KeepPositiveSolution(Solution)
167
168        ListPointIntersection = []
169        for t in Solution:
170            Intersect = Ray.vector * t + Ray.point
171            if Intersect[2] < 0 and self.support._IncludeSupport(Intersect):
172                ListPointIntersection.append(Intersect)
173
174        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
175
176    def get_normal(self, Point):
177        """Return the normal unit vector on the spherical surface at point Point."""
178        Gradient = Point
179        return mgeo.Normalize(-Gradient)
180
181    def get_centre(self):
182        """Return 3D coordinates of the point on the mirror surface at the center of its support."""
183        return np.array([0, 0, -self.radius])
184
185    def get_grid3D(self, NbPoint, **kwargs):
186        """
187        Get grid of points on mirror surface.
188
189        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
190        sampling the support in a number NbPoints of points.
191        """
192        E = "edges" in kwargs and kwargs["edges"]
193        ListCoordXYZ = []
194        contour = int(round(0.1 * NbPoint))
195        contours = self.support._Contour_points(contour, **kwargs)
196        if E:
197            contours, contour_edges = contours
198        ListCoordXY = contours + self.support._get_grid(NbPoint - contour)
199        for k in ListCoordXY:
200            z = -np.sqrt(self.radius**2 - (k[0] ** 2 + k[1] ** 2))
201            ListCoordXYZ.append(np.array([k[0], k[1], z]))
202        if E:
203            return ListCoordXYZ, contour_edges
204        return ListCoordXYZ

208class MirrorParabolic:
209    r"""
210    A paraboloid with vertex at the origin $O=[0,0,0]$ and symmetry axis z:
211    $z = \frac{1}{4f}[x^2 + y^2]$ where $f$ is the focal lenght of the *mother*
212    parabola (i.e. measured from its center at $O$ to the focal point $F$).
213
214    The center of the support is shifted along the x-direction by the off-axis distance $x_c$.
215    This leads to an *effective focal length* $f_\mathrm{eff}$, measured from the shifted center
216    of the support  $P$ to the focal point $F$.
217    It is related to the mother focal length by $f = f_\\mathrm{eff} \cos^2(\alpha/2) $,
218    or equivalently $ p = 2f = f_\mathrm{eff} (1+\\cos\\alpha)$, where $\\alpha$
219    is the off-axis angle, and $p = 2f$ is called the semi latus rectum.
220
221    Another useful relationship is that between the off-axis distance and the resulting
222    off-axis angle: $x_c = 2 f \tan(\alpha/2)$.
223
224
225    ![Illustration of a parabolic mirror.](../docs/parabola.svg)
226
227    Attributes
228    ----------
229        offaxisangle : float
230            Off-axis angle of the parabola. Modifying this also updates p, keeping feff constant.
231            Attention: The off-axis angle must be *given in degrees*, but is stored and will be *returned in radian* !
232
233        feff : float
234            Effective focal length of the parabola in mm. Modifying this also updates p, keeping offaxisangle constant.
235
236        p : float
237            Semi latus rectum of the parabola in mm. Modifying this also updates feff, keeping offaxisangle constant.
238
239        support : ART.ModuleSupport.Support
240
241        type : str 'Parabolic Mirror'.
242
243    Methods
244    -------
245        MirrorParabolic.get_normal(Point)
246
247        MirrorParabolic.get_centre()
248
249        MirrorParabolic.get_grid3D(NbPoints)
250
251    """
252
253    def __init__(self, FocalEffective: float, OffAxisAngle: float, Support):
254        """
255        Initialise an Parabolic mirror.
256
257        Parameters
258        ----------
259            FocalEffective : float
260                Effective focal length of the parabola in mm.
261
262            OffAxisAngle : float
263                Off-axis angle *in degrees* of the parabola.
264
265            Support : ART.ModuleSupport.Support
266
267        """
268        self._offaxisangle = np.deg2rad(OffAxisAngle)
269        self.support = Support
270        self.type = "Parabolic Mirror"
271        self._feff = FocalEffective  # effective focal length
272        self._p = FocalEffective * (
273            1 + np.cos(self.offaxisangle)
274        )  # semi latus rectum, =2*focal length of mother parabola
275
276    @property
277    def offaxisangle(self):
278        return self._offaxisangle
279
280    @offaxisangle.setter
281    def offaxisangle(self, OffAxisAngle):
282        self._offaxisangle = np.deg2rad(OffAxisAngle)  # store (and later return) in radians!
283        self._p = self._feff * (1 + np.cos(self._offaxisangle))  # make sure to always update p
284
285    @property
286    def offaxisangle(self):
287        """Return off-axis angle of parabolic mirror."""
288        return self._offaxisangle
289
290    @offaxisangle.setter
291    def offaxisangle(self, OffAxisAngle):
292        self._offaxisangle = np.deg2rad(OffAxisAngle)
293        self._p = self._feff * (1 + np.cos(self._offaxisangle))  # make sure to always update p
294
295    @property
296    def feff(self):
297        """Get effective focal length."""
298        return self._feff
299
300    @feff.setter
301    def feff(self, FocalEffective):
302        self._feff = FocalEffective
303        self._p = self._feff * (1 + np.cos(self._offaxisangle))  # make sure to always update p
304
305    @property
306    def p(self):
307        """Get simi-latus-rectum."""
308        return self._p
309
310    @p.setter
311    def p(self, SemiLatusRectum):
312        self._p = SemiLatusRectum
313        self._feff = self._p / (1 + np.cos(self._offaxisangle))  # make sure to always update feff
314
315    def _get_intersection(self, Ray):
316        """Return the intersection point between the ray and the parabola."""
317        ux = Ray.vector[0]
318        uy = Ray.vector[1]
319        uz = Ray.vector[2]
320        xA = Ray.point[0]
321        yA = Ray.point[1]
322        zA = Ray.point[2]
323
324        da = ux**2 + uy**2
325        db = 2 * (ux * xA + uy * yA) - 2 * self._p * uz
326        dc = xA**2 + yA**2 - 2 * self._p * zA
327
328        Solution = mgeo.SolverQuadratic(da, db, dc)
329        Solution = mgeo.KeepPositiveSolution(Solution)
330
331        ListPointIntersection = []
332        for t in Solution:
333            Intersect = Ray.vector * t + Ray.point
334            if self.support._IncludeSupport(Intersect - self.get_centre()):
335                ListPointIntersection.append(Intersect)
336
337        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
338
339    def get_normal(self, Point):
340        """Return the normal unit vector on the paraboloid surface at point Point."""
341        Gradient = np.zeros(3)
342        Gradient[0] = -Point[0]
343        Gradient[1] = -Point[1]
344        Gradient[2] = self._p
345        return mgeo.Normalize(Gradient)
346
347    def get_centre(self):
348        """Return 3D coordinates of the point $P$ on the mirror surface at the center of its support."""
349        return np.array(
350            [self.feff * np.sin(self.offaxisangle), 0, self._p * 0.5 - self.feff * np.cos(self.offaxisangle)]
351        )
352
353    def get_grid3D(self, NbPoint, **kwargs):
354        """
355        Get grid of points on mirror surface.
356
357        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
358        sampling the support in a number NbPoints of points.
359        """
360        E = "edges" in kwargs and kwargs["edges"]
361        ListCoordXYZ = []
362        contour = int(round(0.1 * NbPoint))
363        contours = self.support._Contour_points(contour, **kwargs)
364        if E:
365            contours, contour_edges = contours
366        ListCoordXY = contours + self.support._get_grid(NbPoint - contour)
367        xc = self.feff * np.sin(self.offaxisangle)
368        for k in ListCoordXY:
369            z = ((k[0] + xc) ** 2 + k[1] ** 2) / 2 / self._p
370            ListCoordXYZ.append(np.array([k[0] + xc, k[1], z]))
371        if E:
372            return ListCoordXYZ, contour_edges
373        return ListCoordXYZ

377class MirrorToroidal:
378    r"""
379    Toroidal mirror surface with eqn.$(\sqrt{x^2+z^2}-R)^2 + y^2 = r^2$ where $R$ and $r$ the major and minor radii.
380
381    ![Illustration of a toroidal mirror.](toroidal.svg)
382
383    Attributes
384    ----------
385        majorradius : float
386            Major radius of the toroid in mm.
387
388        minorradius : float
389            Minor radius of the toroid in mm.
390
391        support : ART.ModuleSupport.Support
392
393        type : str 'Toroidal Mirror'.
394
395    Methods
396    -------
397        MirrorToroidal.get_normal(Point)
398
399        MirrorToroidal.get_centre()
400
401        MirrorToroidal.get_grid3D(NbPoints)
402
403    """
404
405    def __init__(self, MajorRadius, MinorRadius, Support):
406        """
407        Construct a toroidal mirror.
408
409        Parameters
410        ----------
411            MajorRadius : float
412                Major radius of the toroid in mm.
413
414            MinorRadius : float
415                Minor radius of the toroid in mm.
416
417            Support : ART.ModuleSupport.Support
418
419        """
420        self.majorradius = MajorRadius
421        self.minorradius = MinorRadius
422        self.support = Support
423        self.type = "Toroidal Mirror"
424
425    def _get_intersection(self, Ray):
426        """Return the intersection point between Ray and the toroidal mirror surface."""
427        ux = Ray.vector[0]
428        uz = Ray.vector[2]
429        xA = Ray.point[0]
430        zA = Ray.point[2]
431
432        G = 4.0 * self.majorradius**2 * (ux**2 + uz**2)
433        H = 8.0 * self.majorradius**2 * (ux * xA + uz * zA)
434        I = 4.0 * self.majorradius**2 * (xA**2 + zA**2)
435        J = np.dot(Ray.vector, Ray.vector)
436        K = 2.0 * np.dot(Ray.vector, Ray.point)
437        L = np.dot(Ray.point, Ray.point) + self.majorradius**2 - self.minorradius**2
438
439        a = J**2
440        b = 2 * J * K
441        c = 2 * J * L + K**2 - G
442        d = 2 * K * L - H
443        e = L**2 - I
444
445        Solution = mgeo.SolverQuartic(a, b, c, d, e)
446        Solution = mgeo.KeepPositiveSolution(Solution)
447
448        ListPointIntersection = []
449        for t in Solution:
450            Intersect = Ray.vector * t + Ray.point
451            if Intersect[2] < -self.majorradius and self.support._IncludeSupport(Intersect):  # For realistic mirror
452                ListPointIntersection.append(Intersect)
453
454        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
455
456    def get_normal(self, Point):
457        """Return the normal unit vector on the toroidal surface at point Point."""
458        x = Point[0]
459        y = Point[1]
460        z = Point[2]
461        A = self.majorradius**2 - self.minorradius**2
462
463        Gradient = np.zeros(3)
464        Gradient[0] = 4 * (x**3 + x * y**2 + x * z**2 + x * A) - 8 * x * self.majorradius**2
465        Gradient[1] = 4 * (y**3 + y * x**2 + y * z**2 + y * A)
466        Gradient[2] = 4 * (z**3 + z * x**2 + z * y**2 + z * A) - 8 * z * self.majorradius**2
467
468        return mgeo.Normalize(-Gradient)
469
470    def get_centre(self):
471        """Return 3D coordinates of the point on the mirror surface at the center of its support."""
472        return np.array([0, 0, -self.majorradius - self.minorradius])
473
474    def get_grid3D(self, NbPoint, **kwargs):
475        """
476        Get grid of points on mirror surface.
477
478        Returns list of numpy-arrays containing the 3D-coordinates of points in the mirror surface,
479        sampling the support in a number NbPoints of points.
480        """
481        E = "edges" in kwargs and kwargs["edges"]
482        ListCoordXYZ = []
483        contour = int(round(0.1 * NbPoint))
484        contours = self.support._Contour_points(contour, **kwargs)
485        if E:
486            contours, contour_edges = contours
487        ListCoordXY = contours + self.support._get_grid(NbPoint - contour)
488        for k in ListCoordXY:
489            z = -np.sqrt((np.sqrt(self.minorradius**2 - k[1] ** 2) + self.majorradius) ** 2 - k[0] ** 2)
490            ListCoordXYZ.append(np.array([k[0], k[1], z]))
491        if E:
492            return ListCoordXYZ, contour_edges
493        return ListCoordXYZ

499def ReturnOptimalToroidalRadii(Focal: float, AngleIncidence: float) -> (float, float):
500    """
501    Get optimal parameters for a toroidal mirror.
502
503    Useful helper function to get the optimal major and minor radii for a toroidal mirror to achieve a
504    focal length 'Focal' with an angle of incidence 'AngleIncidence' and with vanishing astigmatism.
505
506    Parameters
507    ----------
508        Focal : float
509            Focal length in mm.
510
511        AngleIncidence : int
512            Angle of incidence in degrees.
513
514    Returns
515    -------
516        OptimalMajorRadius, OptimalMinorRadius : float, float.
517    """
518    AngleIncidenceRadian = AngleIncidence * np.pi / 180
519    OptimalMajorRadius = 2 * Focal * (1 / np.cos(AngleIncidenceRadian) - np.cos(AngleIncidenceRadian))
520    OptimalMinorRadius = 2 * Focal * np.cos(AngleIncidenceRadian)
521    return OptimalMajorRadius, OptimalMinorRadius

525class MirrorEllipsoidal:
526    """
527    Ellipdoidal mirror surface with eqn. $(x/a)^2 + (y/b)^2 + (z/b)^2 = 1$, where $a$ and $b$ are semi major and semi minor axes.
528
529    ![Illustration of a ellipsoidal mirror.](ellipsoid.svg)
530
531    Attributes
532    ----------
533        a : float
534            Semi major axis of the ellipsoid in mm.
535
536        b : float
537            Semi minor axis of the ellipsoid in mm.
538
539        support : ART.ModuleSupport.Support
540
541        type : str 'Ellipsoidal Mirror'.
542
543    Methods
544    -------
545        MirrorEllipsoidal.get_normal(Point)
546
547        MirrorEllipsoidal.get_centre()
548
549        MirrorEllipsoidal.get_grid3D(NbPoints)
550
551    """
552
553    def __init__(self, Support, SemiMajorAxis=None, SemiMinorAxis=None, OffAxisAngle=None, f_object=None, f_image=None):
554        """
555        Generate an ellipsoidal mirror with given parameters.
556
557        Depending on the parameters provided by the vendor, you can either specify:
558            - the Semi Major and Semi Minor axes (default)
559            - The object and image distances
560        You can also specify the Off-Axis angle in degrees. Keep in mind that its value will be stored and returned in radian.
561
562        Parameters
563        ----------
564        Support : TYPE
565            ART.ModuleSupport.Support.
566        SemiMajorAxis : float
567            Semi major axis of the ellipsoid in mm..
568        SemiMinorAxis : float
569            Semi minor axis of the ellipsoid in mm..
570        OffAxisAngle : float
571            Off-axis angle of the mirror in mm. Defined as the angle at the centre of the mirror between the two foci..
572        f_object : float
573            Object focal distance in mm.
574        f_image : float
575            Image focal distance in mm.
576        """
577        self.type = "Ellipsoidal Mirror"
578        self.support = Support
579        self.a = None
580        self.b = None
581        self._offaxisangle = None
582        if SemiMajorAxis is not None and SemiMinorAxis is not None:
583            self.a = SemiMajorAxis
584            self.b = SemiMinorAxis
585        if OffAxisAngle is not None:
586            self._offaxisangle = np.deg2rad(OffAxisAngle)
587            if f_object is not None and f_image is not None:
588                f_o = f_object
589                f_i = f_image
590                foci_sq = f_o**2 + f_i**2 - 2 * f_o * f_i * np.cos(self._offaxisangle)
591                self.a = (f_i + f_o) / 2
592                self.b = np.sqrt(self.a**2 - foci_sq / 4)
593        else:
594            if f_object is not None and f_image is not None:
595                f_o = f_object
596                f_i = f_image
597                if self.a is not None and self.b is not None:
598                    foci = 2 * np.sqrt(self.a**2 - self.b**2)
599                    self._offaxisangle = np.arccos((f_i**2 + f_o**2 - foci**2) / (2 * f_i * f_o))
600
601            elif self.a is not None and self.b is not None:
602                foci = 2 * np.sqrt(self.a**2 - self.b**2)
603                f = self.a
604                self._offaxisangle = np.arccos(1 - foci**2 / (2 * f**2))
605        if self.a is None or self.b is None or self._offaxisangle is None:
606            raise ValueError("Invalid mirror parameters")
607
608    def _get_intersection(self, Ray):
609        """Return the intersection point between Ray and the ellipsoidal mirror surface."""
610        ux, uy, uz = Ray.vector
611        xA, yA, zA = Ray.point
612
613        da = (uy**2 + uz**2) / self.b**2 + (ux / self.a) ** 2
614        db = 2 * ((uy * yA + uz * zA) / self.b**2 + (ux * xA) / self.a**2)
615        dc = (yA**2 + zA**2) / self.b**2 + (xA / self.a) ** 2 - 1
616
617        Solution = mgeo.SolverQuadratic(da, db, dc)
618        Solution = mgeo.KeepPositiveSolution(Solution)
619
620        ListPointIntersection = []
621        C = self.get_centre()
622        for t in Solution:
623            Intersect = Ray.vector * t + Ray.point
624            if Intersect[2] < 0 and self.support._IncludeSupport(Intersect - C):
625                ListPointIntersection.append(Intersect)
626
627        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
628
629    def get_normal(self, Point):
630        """Return the normal unit vector on the ellipsoidal surface at point Point."""
631        Gradient = np.zeros(3)
632
633        Gradient[0] = -Point[0] / self.a**2
634        Gradient[1] = -Point[1] / self.b**2
635        Gradient[2] = -Point[2] / self.b**2
636
637        return mgeo.Normalize(Gradient)

639    def get_centre(self):
640        """Return 3D coordinates of the point on the mirror surface at the center of its support."""
641        foci = 2 * np.sqrt(self.a**2 - self.b**2)
642        h = -foci / 2 / np.tan(self._offaxisangle)
643        R = np.sqrt(foci**2 / 4 + h**2)
644        sign = 1
645        if math.isclose(self._offaxisangle, np.pi / 2):
646            h = 0
647        elif self._offaxisangle > np.pi / 2:
648            h = -h
649            sign = -1
650        a = 1 - self.a**2 / self.b**2
651        b = -2 * h
652        c = self.a**2 + h**2 - R**2
653        z = (-b + sign * np.sqrt(b**2 - 4 * a * c)) / (2 * a)
654        if math.isclose(z**2, self.b**2):
655            return np.array([0, 0, -self.b])
656        x = self.a * np.sqrt(1 - z**2 / self.b**2)
657        centre = np.array([x, 0, sign * z])
658        return centre

661        """

663

665        sampling the support in a number NbPoints of points.
666        """
667        E = "edges" in kwargs and kwargs["edges"]
668        ListCoordXYZ = []
669        contour = int(round(0.1 * NbPoint))
670        contours = self.support._Contour_points(contour, **kwargs)
671        if E:
672            contours, contour_edges = contours
673        ListCoordXY = self.support._get_grid(NbPoint - contour)
674        centre = self.get_centre()
675        for i, k in enumerate(ListCoordXY):
676            x = k[0] + centre[0]
677            y = k[1]
678            sideways = (x / self.a) ** 2 + (y / self.b) ** 2
679            if sideways <= 1:
680                z = -self.b * np.sqrt(1 - sideways)
681                ListCoordXYZ.append(np.array([x, y, z]))
682        new_contour_edges = []
683        for j in contour_edges:
684            new_contour_edges += [[]]
685            for i in j:
686                x = contours[i][0] + centre[0]
687                y = contours[i][1]
688                sideways = (x / self.a) ** 2 + (y / self.b) ** 2
689                if sideways <= 1:
690                    z = -self.b * np.sqrt(1 - sideways)
691                    ListCoordXYZ.append(np.array([x, y, z]))
692                    new_contour_edges[-1] += [len(ListCoordXYZ) - 1]
693        if E:

A paraboloid with vertex at the origin $O=[0,0,0]$ and symmetry axis z: $z = \frac{1}{4f}[x^2 + y^2]$ where $f$ is the focal lenght of the mother parabola (i.e. measured from its center at $O$ to the focal point $F$).

The center of the support is shifted along the x-direction by the off-axis distance $x_c$. This leads to an effective focal length $f_\mathrm{eff}$, measured from the shifted center of the support $P$ to the focal point $F$. It is related to the mother focal length by $f = f_\mathrm{eff} \cos^2(\alpha/2) $, or equivalently $ p = 2f = f_\mathrm{eff} (1+\cos\alpha)$, where $\alpha$ is the off-axis angle, and $p = 2f$ is called the semi latus rectum.

Another useful relationship is that between the off-axis distance and the resulting off-axis angle: $x_c = 2 f \tan(\alpha/2)$.
