 24class OpticalChain:
 25    """ 
 26    The OpticalChain represents the whole optical setup to be simulated:
 27    Its main attributes are a list source-[Rays](ModuleOpticalRay.html) and 
 28    a list of successive [OpticalElements](ModuleOpticalElement.html).
 29    
 30    The method OpticalChain.get_output_rays() returns an associated list of lists of
 31    [Rays](ModuleOpticalRay.html), each calculated by ray-tracing from one
 32    [OpticalElement](ModuleOpticalElement.html) to the next.
 33    So OpticalChain.get_output_rays()[i] is the bundle of [Rays](ModuleOpticalRay.html) *after*
 34    optical_elements[i].
 35    
 36    The string "description" can contain a short description of the optical setup, or similar notes.
 37    
 38    The OpticalChain can be visualized quickly with the method OpticalChain.quickshow(),
 39    and more nicely with OpticalChain.render().
 40    
 41    The class also provides methods for (mis-)alignment of the source-[Ray](ModuleOpticalRay.html)-bundle and the
 42    [OpticalElements](ModuleOpticalElement.html), as well as methods for producing
 43    a list of OpticalChain-objects containing variations of itself.
 44
 45    Attributes
 46    ----------
 47        source_rays : list[mray.Ray]
 48            List of source rays, which are to be traced.
 49            
 50        optical_elements : list[moe.OpticalElement]
 51            List of successive optical elements. 
 52        
 53        description : str
 54            A string to describe the optical setup.
 55    
 56        loop_variable_name : str
 57            A string naming a parameter that is varied in a list of OpticalChain-objects,
 58            which is useful when looping over variations of an initial configuration.
 59            
 60        loop_variable_value : float
 61            The value of that varied parameter, which is useful when looping over
 62            variations of an initial configuration.
 63
 64    Methods
 65    ----------
 66        copy_chain()

 70        quickshow()

219    def shift_source(self, axis: (str, np.ndarray), distance: float):
220        """
221        Shift source ray bundle by distance (in mm) along the 'axis' specified as
222        a lab-frame vector (numpy-array of length 3) or as one of the strings
223        "vert", "horiz", or "random".
224        
225        In the latter case, the reference is the incidence plane of the first
226        non-normal-incidence mirror after the source. If there is none, you will
227        be asked to rather specify the axis as a 3D-numpy-array.
228        
229        axis = "vert" means the source position is shifted along the axis perpendicular
230        to that incidence plane, i.e. "vertically" away from the former incidence plane..
231        
232        axis = "horiz" means the source direciton is rotated about an axis in that
233        incidence plane and perpendicular to the current source direction,
234        i.e. "horizontally" in the incidence plane, but retaining the same distance
235        of source and first optical element.
236        
237        axis = "random" means the the source direction shifted in a random direction
238        within in the plane perpendicular to the current source direction,
239        e.g. simulating a fluctuation of hte transverse source position.
240        
241        Parameters
242        ----------
243            axis : np.ndarray or str
244                Shift axis, specified either as a 3D lab-frame vector or as one
245                of the strings "vert", "horiz", or "random".
246                
247            distance : float
248                Shift distance in mm.
249
250        Returns
251        -------
252            Nothing, just modifies the property 'source_rays'.
253        """
254        if type(distance) not in [int, float, np.float64]:
255            raise ValueError('The "distance"-argument must be an int or float number.')
256        
257        central_ray_vector = mp.FindCentralRay(self.source_rays).vector
258        mirror_indcs = [i for i, OE in enumerate(self.optical_elements) if 'Mirror' in OE.type.type]
259        
260        OEnormal = None
261        for i in mirror_indcs:
262            ith_OEnormal = self.optical_elements[i].normal
263            if np.linalg.norm(np.cross(central_ray_vector, ith_OEnormal)) > 1e-10:
264                OEnormal = ith_OEnormal
265                break
266        if OEnormal is None:
267            raise Exception("There doesn't seem to be a non-normal-incidence mirror in this optical chain, \
268                            so you should rather give 'axis' as a numpy-array of length 3.")
269        
270        if type(axis) == np.ndarray and len(axis) == 3:
271            translation_vector = axis
272        else:
273            perp_axis = np.cross(central_ray_vector,OEnormal)
274            horiz_axis = np.cross(perp_axis, central_ray_vector)
275            
276            if axis == "vert":
277                translation_vector = perp_axis
278            elif axis == "horiz":
279                translation_vector = horiz_axis
280            elif axis == "random":
281                translation_vector = np.random.uniform(low=-1, high=1, size=1)*perp_axis + np.random.uniform(low=-1, high=1, size=1)*horiz_axis
282            else:
283                raise ValueError('The shift direction must be specified by "axis" as one of ["vert", "horiz", "random"].')
284            
285        self.source_rays = mgeo.TranslationRayList(self.source_rays, distance*mgeo.Normalize(translation_vector))

288    def tilt_source(self, axis: (str, np.ndarray), angle: float):
289        """ 
290        Rotate source ray bundle by angle around an axis, specified as
291        a lab-frame vector (numpy-array of length 3) or as one of the strings
292        "in_plane", "out_plane" or "random" direction.
293        
294        In the latter case, the function considers the incidence plane of the first
295        non-normal-incidence mirror after the source. If there is none, you will
296        be asked to rather specify the axis as a 3D-numpy-array.
297        
298        axis = "in_plane" means the source direction is rotated about an axis
299        perpendicular to that incidence plane, which tilts the source
300        "horizontally" in the same plane.
301        
302        axis = "out_plane" means the source direciton is rotated about an axis
303        in that incidence plane and perpendicular to the current source direction,
304        which tilts the source "vertically" out of the former incidence plane.
305        
306        axis = "random" means the the source direction is tilted in a random direction,
307        e.g. simulating a beam pointing fluctuation.
308        
309        Attention, "angle" is given in deg, so as to remain consitent with the
310        conventions of other functions, although pointing is mostly talked about 
311        in mrad instead.
312        
313        Parameters
314        ----------
315            axis : np.ndarray or str
316                Shift axis, specified either as a 3D lab-frame vector or as one
317                of the strings "in_plane", "out_plane", or "random".
318                
319            angle : float
320                Rotation angle in degree.
321
322        Returns
323        -------
324            Nothing, just modifies the property 'source_rays'.
325        """
326        if type(angle) not in [int, float, np.float64]:
327            raise ValueError('The "angle"-argument must be an int or float number.')
328        
329        central_ray_vector = mp.FindCentralRay(self.source_rays).vector
330        mirror_indcs = [i for i, OE in enumerate(self.optical_elements) if 'Mirror' in OE.type.type]
331        
332        OEnormal = None
333        for i in mirror_indcs:
334            ith_OEnormal = self.optical_elements[i].normal
335            if np.linalg.norm(np.cross(central_ray_vector, ith_OEnormal)) > 1e-10:
336                OEnormal = ith_OEnormal
337                break
338        if OEnormal is None:
339            raise Exception("There doesn't seem to be a non-normal-incidence mirror in this optical chain, \
340                            so you should rather give 'axis' as a numpy-array of length 3.")
341        
342        if type(axis) == np.ndarray and len(axis) == 3:
343            rot_axis = axis
344        else:
345            rot_axis_in = np.cross(central_ray_vector,OEnormal)
346            rot_axis_out = np.cross(rot_axis_in, central_ray_vector)
347            if axis == "in_plane":
348                rot_axis = rot_axis_in
349            elif axis == "out_plane":
350                rot_axis = rot_axis_out
351            elif axis == "random":
352                rot_axis = np.random.uniform(low=-1, high=1, size=1)*rot_axis_in + np.random.uniform(low=-1, high=1, size=1)*rot_axis_out
353            else:
354                raise ValueError('The tilt axis must be specified by as one of ["in_plane", "out_plane", "random"] or as a numpy-array of length 3.')
355             
356        self.source_rays = mgeo.RotationAroundAxisRayList(self.source_rays, rot_axis, np.deg2rad(angle))

359    def get_source_loop_list(self, axis :str, loop_variable_values : np.ndarray):
360        """
361        Produces a list of OpticalChain-objects, which are all variations of this
362        instance by moving the source-ray-bundle.
363        The variation is specified by axis as one of
364        ["tilt_in_plane", "tilt_out_plane", "tilt_random", "shift_vert", "shift_horiz", "shift_random"],
365        by the values given in the list or numpy-array "loop_variable_values", e.g. np.linspace(start, stop, number). 
366        This list can then be looped over by ARTmain.
367        
368        Parameters
369        ----------
370            axis : np.ndarray or str
371                Shift/Rotation axis for the source-modification, specified either
372                as a 3D lab-frame vector or as one of the strings 
373                ["tilt_in_plane", "tilt_out_plane", "tilt_random",\
374                 "shift_vert", "shift_horiz", "shift_random"].
375                
376             loop_variable_values : list or np.ndarray
377                Values of the shifts (mm) or rotations (deg).
378
379        Returns
380        -------
381            OpticalChainList : list[OpticalChain]
382        """
383        if axis not in ["tilt_in_plane", "tilt_out_plane", "tilt_random", "shift_vert", "shift_horiz", "shift_random", "all_random"]:
384            raise ValueError('For automatic loop-list generation, the axis must be one of ["tilt_in_plane", "tilt_out_plane", "tilt_random", "shift_vert", "shift_horiz", "shift_random"].')
385        if type(loop_variable_values) not in [list, np.ndarray]:
386            raise ValueError('For automatic loop-list generation, the loop_variable_values must be a list or a numpy-array.')
387        
388        loop_variable_name_strings = {
389            'tilt_in_plane' : 'source tilt in-plane (deg)',
390            'tilt_out_plane': 'source tilt out-of-plane (deg)',
391            'tilt_random'   : 'source tilt random axis (deg)',
392            'shift_vert'    : 'source shift vertical (mm)',
393            'shift_horiz'   : 'source shift horizontal (mm)',
394            'shift_random'  : 'source shift random-direction (mm)'
395            }
396        loop_variable_name = loop_variable_name_strings[axis]
397        
398        OpticalChainList = []
399        for x in loop_variable_values:
400            # always start with a fresh deep-copy the AlignedOpticalChain, to then modify it and append it to the list
401            ModifiedOpticalChain = self.copy_chain()
402            ModifiedOpticalChain.loop_variable_name = loop_variable_name
403            ModifiedOpticalChain.loop_variable_value = x
404            
405            if axis in ["tilt_in_plane", "tilt_out_plane", "tilt_random"]:
406                ModifiedOpticalChain.tilt_source(axis[5:], x)
407            elif axis in ["shift_vert", "shift_horiz", "shift_random"]:
408                ModifiedOpticalChain.shift_source(axis[6:], x)
409            
410            #append the modified optical chain to the list
411            OpticalChainList.append(ModifiedOpticalChain)
412            
413        return OpticalChainList

 27    The OpticalChain represents the whole optical setup to be simulated:
 28    Its main attributes are a list source-[Rays](ModuleOpticalRay.html) and 
 29    a list of successive [OpticalElements](ModuleOpticalElement.html).

220    def shift_source(self, axis: (str, np.ndarray), distance: float):
221        """
222        Shift source ray bundle by distance (in mm) along the 'axis' specified as
223        a lab-frame vector (numpy-array of length 3) or as one of the strings
224        "vert", "horiz", or "random".
225        
226        In the latter case, the reference is the incidence plane of the first
227        non-normal-incidence mirror after the source. If there is none, you will
228        be asked to rather specify the axis as a 3D-numpy-array.
229        
230        axis = "vert" means the source position is shifted along the axis perpendicular
231        to that incidence plane, i.e. "vertically" away from the former incidence plane..
232        
233        axis = "horiz" means the source direciton is rotated about an axis in that
234        incidence plane and perpendicular to the current source direction,
235        i.e. "horizontally" in the incidence plane, but retaining the same distance
236        of source and first optical element.
237        
238        axis = "random" means the the source direction shifted in a random direction
239        within in the plane perpendicular to the current source direction,
240        e.g. simulating a fluctuation of hte transverse source position.
241        
242        Parameters
243        ----------
244            axis : np.ndarray or str
245                Shift axis, specified either as a 3D lab-frame vector or as one
246                of the strings "vert", "horiz", or "random".
247                
248            distance : float
249                Shift distance in mm.
250
251        Returns
252        -------
253            Nothing, just modifies the property 'source_rays'.
254        """
255        if type(distance) not in [int, float, np.float64]:
256            raise ValueError('The "distance"-argument must be an int or float number.')
257        
258        central_ray_vector = mp.FindCentralRay(self.source_rays).vector
259        mirror_indcs = [i for i, OE in enumerate(self.optical_elements) if 'Mirror' in OE.type.type]
260        
261        OEnormal = None
262        for i in mirror_indcs:
263            ith_OEnormal = self.optical_elements[i].normal
264            if np.linalg.norm(np.cross(central_ray_vector, ith_OEnormal)) > 1e-10:
265                OEnormal = ith_OEnormal
266                break
267        if OEnormal is None:
268            raise Exception("There doesn't seem to be a non-normal-incidence mirror in this optical chain, \
269                            so you should rather give 'axis' as a numpy-array of length 3.")
270        
271        if type(axis) == np.ndarray and len(axis) == 3:
272            translation_vector = axis
273        else:
274            perp_axis = np.cross(central_ray_vector,OEnormal)
275            horiz_axis = np.cross(perp_axis, central_ray_vector)
276            
277            if axis == "vert":
278                translation_vector = perp_axis
279            elif axis == "horiz":
280                translation_vector = horiz_axis
281            elif axis == "random":
282                translation_vector = np.random.uniform(low=-1, high=1, size=1)*perp_axis + np.random.uniform(low=-1, high=1, size=1)*horiz_axis
283            else:
284                raise ValueError('The shift direction must be specified by "axis" as one of ["vert", "horiz", "random"].')
285            
286        self.source_rays = mgeo.TranslationRayList(self.source_rays, distance*mgeo.Normalize(translation_vector))

289    def tilt_source(self, axis: (str, np.ndarray), angle: float):
290        """ 
291        Rotate source ray bundle by angle around an axis, specified as

306

319                
320            angle : float
321                Rotation angle in degree.
322
323        Returns
324        -------
325            Nothing, just modifies the property 'source_rays'.
326        """
327        if type(angle) not in [int, float, np.float64]:
328            raise ValueError('The "angle"-argument must be an int or float number.')
329        
330        central_ray_vector = mp.FindCentralRay(self.source_rays).vector
331        mirror_indcs = [i for i, OE in enumerate(self.optical_elements) if 'Mirror' in OE.type.type]

333        OEnormal = None
334        for i in mirror_indcs:
335            ith_OEnormal = self.optical_elements[i].normal

338                break

344            rot_axis = axis
345        else:

347            rot_axis_out = np.cross(rot_axis_in, central_ray_vector)
348            if axis == "in_plane":
349                rot_axis = rot_axis_in
350            elif axis == "out_plane":
351                rot_axis = rot_axis_out
352            elif axis == "random":
353                rot_axis = np.random.uniform(low=-1, high=1, size=1)*rot_axis_in + np.random.uniform(low=-1, high=1, size=1)*rot_axis_out
354            else:
355                raise ValueError('The tilt axis must be specified by as one of ["in_plane", "out_plane", "random"] or as a numpy-array of length 3.')
356             
357        self.source_rays = mgeo.RotationAroundAxisRayList(self.source_rays, rot_axis, np.deg2rad(angle))

The OpticalChain represents the whole optical setup to be simulated: Its main attributes are a list source-Rays and a list of successive OpticalElements.

The method OpticalChain.get_output_rays() returns an associated list of lists of Rays, each calculated by ray-tracing from one OpticalElement to the next. So OpticalChain.get_output_rays()[i] is the bundle of Rays after optical_elements[i].

The string "description" can contain a short description of the optical setup, or similar notes.

The OpticalChain can be visualized quickly with the method OpticalChain.quickshow(), and more nicely with OpticalChain.render().

The class also provides methods for (mis-)alignment of the source-Ray-bundle and the OpticalElements, as well as methods for producing a list of OpticalChain-objects containing variations of itself.
