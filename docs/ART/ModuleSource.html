ART.ModuleSource

Provides functions that create lists of Ray-objects that can serve as light sources for the start of the ray tracing calculation.

These are called by 'ARTmain' according to the values in the SourceProperties-dictionary filled in the CONFIG-scripts..

 21def _Cone(Angle : float, NbRays : int, Wavelength = None):
 22    """
 23    Return a list of rays filling a cone according to Vogel's spiral.
 24     
 25    Parameters
 26    ----------
 27        Angle : float
 28            Cone apex *half*-angle in radians.
 29        
 30        NbRays : int
 31            Number of rays in the list.
 32            
 33        Wavelength : float, optional
 34            Wavelength of the light, set as attribute to all rays. 
 35    """
 36    Height = 1
 37    Radius = Height * np.tan(Angle)
 38    
 39    MatrixXY = mgeo.SpiralVogel(NbRays, Radius)
 40    
 41    RayList = []
 42    
 43    for k in range(NbRays):
 44        x = MatrixXY[k,0]
 45        y = MatrixXY[k,1]
 46        RayList.append(mray.Ray(np.array([0,0,0]), np.array([x,y,Height]), Number=k, Wavelength = Wavelength))
 47  
 48    return RayList

 52def PointSource(S : np.ndarray, Axis : np.ndarray, Divergence : float, NbRays : int, Wavelength = None):
 53    """
 54    Return a list of rays simulating rays from a point source.
 55   
 56    ![Illustration of a point source.](../docs/pointsource.svg)
 57    
 58    Parameters
 59    ----------
 60        S : np.ndarray
 61            Coordinate vector of the source point (in mm).
 62            
 63        Axis : np.ndarray
 64            Axis vector of the cone (main direction of the rays)
 65        
 66        Divergence : float
 67            Cone apex *half*-angle in *radians* (similar to gaussian beam).
 68        
 69        NbRays : int
 70            Number of rays in the list.
 71            
 72        Wavelength : float, optional
 73            Wavelength of the light, set as attribute to all rays. 
 74    
 75    """
 76    RayList = _Cone(Divergence, NbRays, Wavelength = Wavelength)
 77    RayList = mgeo.RotationRayList(RayList, np.array([0,0,1]), Axis)
 78    RayList = mgeo.TranslationRayList(RayList, S)
 79    return RayList

 83def ExtendedSource(S : np.ndarray, Axis : np.ndarray, Diameter : float, Divergence : float, NbRays : int, Wavelength = None):
 84    """
 85    Return a list of rays simulating rays from an extended array of point sources, distributed over a disk with Diameter.
 86    
 87    Parameters
 88    ----------
 89        S : np.ndarray
 90            Coordinate vector of the source point (in mm).
 91            
 92        Axis : np.ndarray
 93            Axis vector of the cone (main direction of the rays)
 94            
 95        Diameter : float
 96            Diameter of the extended source
 97        
 98        Divergence : float
 99            Cone apex *half*-angle in *radians* (similar to gaussian beam).
100        
101        NbRays : int
102            Number of rays in the list.
103            
104        Wavelength : float, optional
105            Wavelength of the light, set as attribute to all rays. 
106    
107    """
108    NbPointSources = max(5, int(250*Diameter))
109    NbPointSources = min(NbPointSources, 100)
110    
111    MatrixXY = mgeo.SpiralVogel(NbPointSources, Diameter/2) #the positions of the point sources
112    
113    NbRaysPerPointSource = max(100, int(NbRays/NbPointSources))
114    RayList = []
115    PointSourceRayList = _Cone(Divergence, NbRaysPerPointSource, Wavelength = Wavelength)
116    for k in range(NbPointSources):
117        ShiftedPointSourceRayList = mgeo.TranslationRayList(PointSourceRayList, [MatrixXY[k,0],MatrixXY[k,1],0])
118        for l in range(NbRaysPerPointSource):
119            ShiftedPointSourceRayList[l]._number += k*NbRaysPerPointSource #we allow ourselves exceptionally to modify the "internal" _number attribute
120        RayList.extend(ShiftedPointSourceRayList)    
121    
122    RayList = mgeo.RotationRayList(RayList, np.array([0,0,1]), Axis)
123    RayList = mgeo.TranslationRayList(RayList, S)
124    return RayList

128def PlaneWaveDisk(Centre : np.ndarray, Axis : np.ndarray, Radius : float, NbRays : int, Wavelength = None):
129    """
130    Return a list of rays, all parallel and distributed over a round cross section according to a Vogel-spiral (see illustration). Simulating a 'round' collimated beam.
131    
132    ![Illustration of a Vogel spiral.](../docs/doc_VogelSpiral.png)
133    
134    Parameters
135    ----------
136        Centre : np.ndarray
137            Coordinate vector of the center of the source disk (in mm).
138            
139        Axis : np.ndarray
140            Axis vector of the the plane wave (vector of all rays)
141        
142        Radius : float
143            Radius of the source ray-bundle.
144        
145        NbRays : int
146            Number of rays in the list.
147            
148        Wavelength : float, optional
149            Wavelength of the light, set as attribute to all rays. 
150    
151    """
152    MatrixXY = mgeo.SpiralVogel(NbRays, Radius)
153    RayList = []
154    num=0
155    for k in range(NbRays-1):
156        x = MatrixXY[k,0]
157        y = MatrixXY[k,1]
158        RayList.append(mray.Ray(np.array([x,y,0]),np.array([0,0,1]), Number=num, Wavelength = Wavelength))
159        num = num + 1
160    RayList = mgeo.RotationRayList(RayList, np.array([0,0,1]), Axis)
161    RayList = mgeo.TranslationRayList(RayList, Centre)
162    return RayList

165def PlaneWaveSquare(Centre : np.ndarray, Axis: np.ndarray, SideLength : float, NbRays : int, Wavelength = None):
166    """
167    Return a list of rays, all parallel and distributed over a square cross section awith SideLength. Simulating a 'square' collimated beam.
168    
169    Parameters
170    ----------
171        Centre : np.ndarray
172            Coordinate vector of the center of the source disk (in mm).
173            
174        Axis : np.ndarray
175            Axis vector of the the plane wave (vector of all rays)
176        
177        SideLength : float
178            Side length of the square source ray-bundle.
179        
180        NbRays : int
181            Number of rays in the list.
182            
183        Wavelength : float, optional
184            Wavelength of the light, set as attribute to all rays. 
185    
186    """
187    RayList = []
188    x = np.linspace(-SideLength/2,SideLength/2,int(np.sqrt(NbRays)))
189    y = np.linspace(-SideLength/2,SideLength/2,int(np.sqrt(NbRays)))
190    RayList.append(mray.Ray(np.array([0,0,0]),np.array([0,0,1]), Number=0))
191    num = 1
192    for i in x:
193        for j in y:
194            if abs(x) >1e-4 and abs(y)>1e-4:
195                RayList.append(mray.Ray(np.array([i,j,0]),np.array([0,0,1]), Number=num, Wavelength = Wavelength))
196                num += 1
197    RayList = mgeo.RotationRayList(RayList, np.array([0,0,1]), Axis)
198    RayList = mgeo.TranslationRayList(RayList, Centre)
199    return RayList

213def ApplyGaussianIntensityToRayList(RayList, IntensityFraction = 1/np.e**2):
214    """
215    Apply a Gaussain intensity profile to a ray list, with intensity =1 at the center and falling
216    to 'IntensityFraction' at the edge (arbitrary units).
217    
218    Parameters
219    ----------
220        RayList : list of Ray-objects
221            Coordinate vector of the center of the source disk (in mm).
222            
223        IntensityFraction : float, optional
224            Relative intensity in (0,1), reached at the edge of the ray bundle. Default is 1/e^2.
225        
226    """
227    if IntensityFraction >=1 or IntensityFraction <=0:
228        #raise ValueError('IntensityFraction should be between 0 and 1!')
229        print("When applying a Gaussian intensity profile to a ray list, the IntensityFraction should be between 0 and 1! I'm setting it to 1/e^2.")
230        IntensityFraction = 1/np.e**2
231        
232    Axis = mp.FindCentralRay(RayList).vector
233    Divergence = 0
234    for l in RayList:
235        #find the largest angle with the central ray out of all the rays in the list
236        Divergence = max(Divergence,mgeo.AngleBetweenTwoVectors(Axis, l.vector))
237        
238    if Divergence >1e-12: #we're dealing with a point source and not a plane wave, so we can apply Gaussian intensity profile as function of angle
239        for k in RayList:
240            Angle = mgeo.AngleBetweenTwoVectors(Axis,k.vector)
241            Intensity = np.exp(-2*(np.tan(Angle)/Divergence)**2 *-0.5*np.log(IntensityFraction)) 
242            k.intensity = Intensity
243    else:  #otherwise we apply it as function of ray distance from the Axis
244        MaxDist = 0
245        for l in RayList:
246            MaxDist = max(MaxDist,np.linalg.norm(l.point))
247        for k in RayList: 
248            Intensity = np.exp(-2*(np.linalg.norm(k.point)/MaxDist)**2 *-0.5*np.log(IntensityFraction)) 
249            k.intensity = Intensity
250    
251    return RayList
