401            Major radius of the toroid in mm.

404            Minor radius of the toroid in mm.

417

419
420    def __init__(self, MajorRadius, MinorRadius, Support):
421        """
422        Construct a toroidal mirror.
423
424        Parameters
425        ----------
426            MajorRadius : float
427                Major radius of the toroid in mm.
428
429            MinorRadius : float
430                Minor radius of the toroid in mm.
431
432            Support : ART.ModuleSupport.Support
433
434        """
435        self.majorradius = MajorRadius
436        self.minorradius = MinorRadius
437        self.support = Support
438        self.type = "Toroidal Mirror"
439
440    def _get_intersection(self, Ray):
441        """Return the intersection point between Ray and the toroidal mirror surface."""
442        ux = Ray.vector[0]
443        uz = Ray.vector[2]
444        xA = Ray.point[0]
445        zA = Ray.point[2]
446
447        G = 4.0 * self.majorradius**2 * (ux**2 + uz**2)
448        H = 8.0 * self.majorradius**2 * (ux * xA + uz * zA)
449        I = 4.0 * self.majorradius**2 * (xA**2 + zA**2)
450        J = np.dot(Ray.vector, Ray.vector)
451        K = 2.0 * np.dot(Ray.vector, Ray.point)
452        L = (
453            np.dot(Ray.point, Ray.point)
454            + self.majorradius**2
455            - self.minorradius**2
456        )
457
458        a = J**2
459        b = 2 * J * K
460        c = 2 * J * L + K**2 - G
461        d = 2 * K * L - H
462        e = L**2 - I
463
464        Solution = mgeo.SolverQuartic(a, b, c, d, e)
465        Solution = mgeo.KeepPositiveSolution(Solution)
466
467        ListPointIntersection = []
468        for t in Solution:
469            Intersect = Ray.vector * t + Ray.point
470            if Intersect[2] < -self.majorradius and self.support._IncludeSupport(
471                Intersect
472            ):  # For realistic mirror
473                ListPointIntersection.append(Intersect)
474
475        return _IntersectionRayMirror(Ray.point, ListPointIntersection)
476
477    def get_normal(self, Point):
478        """Return the normal unit vector on the toroidal surface at point Point."""
479        x = Point[0]
