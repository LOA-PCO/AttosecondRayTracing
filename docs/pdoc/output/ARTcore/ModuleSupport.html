472def find_hull(points):
473    # start from leftmost point
474    current_point = min(range(len(points)), key=lambda i: points[i][0])
475    # initialize hull with current point
476    hull = [current_point]
477    # initialize list of linked points
478    linked = []
479    # continue until all points have been linked
480    while len(linked) < len(points) - 1:
481        # initialize minimum distance and closest point
482        min_distance = math.inf
483        closest_point = None
484        # find closest unlinked point to current point
485        for i, point in enumerate(points):
486            if i not in linked:
487                distance = math.dist(points[current_point], point)
488                if distance < min_distance:
489                    min_distance = distance
490                    closest_point = i
491        # add closest point to hull and linked list
492        hull.append(closest_point)
493        linked.append(closest_point)
494        # update current point
495        current_point = closest_point
496    # add link between last point and first point
497    hull.append(hull[0])
498    # convert hull to a list of pairs of indices
499    indices = [[hull[i], hull[i + 1]] for i in range(len(hull) - 1)]
500    return indices
def gen_rectangle_contour(dimX, dimY, NbPoints): 
503def gen_rectangle_contour(dimX, dimY, NbPoints):
504    # calculate number of points on each side
505    nX = math.ceil(dimX / (dimX + dimY) * NbPoints)
506    nY = NbPoints - nX
507    # calculate distance between points on each side
508    dX = dimX / (nX - 1)
509    dY = dimY / (nY - 1)
510    # generate points on top side
511    top_points = [(i * dX - dimX / 2, dimY / 2) for i in range(nX)]
512    # generate points on right side
513    right_points = [(dimX / 2, dimY / 2 - i * dY) for i in range(1, nY)]
514    # generate points on bottom side
515    bottom_points = [(dimX / 2 - i * dX, -dimY / 2) for i in range(1, nX)]
516    # generate points on left side
517    left_points = [(-dimX / 2, -dimY / 2 + i * dY) for i in range(1, nY - 1)]
518    # concatenate points in counterclockwise order
519    points = top_points + right_points + bottom_points + left_points
520    return np.array(points)

523def gen_circle_contour(radius, NbPoints):
524    # calculate angle between points
525    if NbPoints == 0:
526        return np.array([])
527    else:    
528        angle = 2 * math.pi / NbPoints
529    # generate points
530    points = [(radius * math.cos(i * angle), radius * math.sin(i * angle)) for i in range(NbPoints)]
531    return np.array(points)
